27 void CodegenEnv::startEmit(OpOperand *so, unsigned lv) {
28  assert(sparseOut == nullptr && insChain == nullptr &&
29  "must only start emitting once");
30  sparseOut = so;
31  outerParNest = lv;
32  if (sparseOut) {
33  insChain = sparseOut->get();
34  latticeMerger.setHasSparseOut(true);
35  }
36  // Initialize loop emitter.
37  SmallVector<Value> tensors;
38  for (OpOperand &t : linalgOp->getOpOperands())
39  tensors.push_back(t.get());
40  loopEmitter.initialize(tensors,
41  StringAttr::get(linalgOp.getContext(),
42  linalg::GenericOp::getOperationName()),
43  /*hasOutput=*/true,
44  /*isSparseOut=*/sparseOut != nullptr, topSort);
45 }

47 Optional<Operation *> CodegenEnv::genLoopBoundary(
48  function_ref<Optional<Operation *>(MutableArrayRef<Value> parameters)>
49  callback) {
50  SmallVector<Value> params;
51  if (isReduc())
52  params.push_back(redVal);
53  if (isExpand())
54  params.push_back(expCount);
55  if (insChain != nullptr)
56  params.push_back(insChain);
57  auto r = callback(params); // may update parameters
58  unsigned i = 0;
59  if (isReduc())
60  updateReduc(params[i++]);
61  if (isExpand())
62  updateExpandCount(params[i++]);
63  if (insChain != nullptr)
64  updateInsertionChain(params[i]);
65  return r;
66 }
