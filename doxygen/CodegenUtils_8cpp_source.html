24 /// If the tensor is a sparse constant, generates and returns the pair of
25 /// the constants for the indices and the values.
26 static Optional<std::pair<Value, Value>>
27 genSplitSparseConstant(OpBuilder &builder, Location loc, Value tensor) {
28  if (auto constOp = tensor.getDefiningOp<arith::ConstantOp>()) {
29  if (auto attr = constOp.getValue().dyn_cast<SparseElementsAttr>()) {
30  DenseElementsAttr indicesAttr = attr.getIndices();
31  Value indices = builder.create<arith::ConstantOp>(loc, indicesAttr);
32  DenseElementsAttr valuesAttr = attr.getValues();
33  Value values = builder.create<arith::ConstantOp>(loc, valuesAttr);
34  return std::make_pair(indices, values);
35  }
36  }
37  return {};
38 }

40 /// Generates the code to copy the index at indices[ivs] to ind, and return
41 /// the value at value[ivs].
42 static Value genIndexAndValueForSparse(OpBuilder &builder, Location loc,
43  Value indices, Value values,
44  SmallVectorImpl<Value> &indicesArray,
45  ValueRange ivs, unsigned rank) {
46  for (unsigned i = 0; i < rank; i++) {
47  Value idx = constantIndex(builder, loc, i);
48  Value val = builder.create<tensor::ExtractOp>(loc, indices,
49  ValueRange{ivs[0], idx});
50  val = builder.create<arith::IndexCastOp>(loc, builder.getIndexType(), val);
51  // builder.create<memref::StoreOp>(loc, val, ind, idx);
52  indicesArray.push_back(val);
53  }
54  return builder.create<tensor::ExtractOp>(loc, values, ivs[0]);
55 }

57 /// Generates the code to read the value from tensor[ivs], and conditionally
58 /// stores the indices ivs to the memory in ind. The generated code looks like
59 /// the following and the insertion point after this routine is inside the
60 /// if-then branch behind the assignment to ind. This is to ensure that the
61 /// code that uses the ind, such as an addEltX call generated after, is inside
62 /// the if-then branch.
63 /// if (tensor[ivs] != 0)
64 /// ind = ivs
65 static Value genIndexAndValueForDense(OpBuilder &builder, Location loc,
66  Value tensor,
67  SmallVectorImpl<Value> &indicesArray,
68  ValueRange ivs) {
69  Value val = genValueForDense(builder, loc, tensor, ivs);
70  indicesArray.append(ivs.begin(), ivs.end());
71  return val;
72 }

220 Value mlir::sparse_tensor::genIsNonzero(OpBuilder &builder, mlir::Location loc,
221  Value v) {
222  Type tp = v.getType();
223  Value zero = constantZero(builder, loc, tp);
224  if (tp.isa<FloatType>())
225  return builder.create<arith::CmpFOp>(loc, arith::CmpFPredicate::UNE, v,
226  zero);
227  if (tp.isIntOrIndex())
228  return builder.create<arith::CmpIOp>(loc, arith::CmpIPredicate::ne, v,
229  zero);
230  if (tp.dyn_cast<ComplexType>())
231  return builder.create<complex::NotEqualOp>(loc, v, zero);
232  llvm_unreachable("Non-numeric type");
233 }

235 void mlir::sparse_tensor::genReshapeDstShape(
236  Location loc, PatternRewriter &rewriter, SmallVectorImpl<Value> &dstShape,
237  ArrayRef<Value> srcShape, ArrayRef<int64_t> staticDstShape,
238  ArrayRef<ReassociationIndices> reassociation) {
239  // Collapse shape.
240  if (reassociation.size() < srcShape.size()) {
241  unsigned start = 0;
242  for (const auto &map : llvm::enumerate(reassociation)) {
243  auto dstDim = constantIndex(rewriter, loc, 1);
244  for (unsigned i = start; i < start + map.value().size(); i++) {
245  dstDim = rewriter.create<arith::MulIOp>(loc, dstDim, srcShape[i]);
246  }
247  dstShape.push_back(dstDim);
248  start = start + map.value().size();
249  }
250  assert(start == srcShape.size());
251  return;
252  }
253 
254  // Expand shape.
255  assert(reassociation.size() == srcShape.size());
256  unsigned start = 0;
257  // Expand the i-th dimension in srcShape.
258  for (unsigned i = 0, size = srcShape.size(); i < size; i++) {
259  const auto &map = reassociation[i];
260  auto srcDim = srcShape[i];
261  // Iterate through dimensions expanded from the i-th dimension.
262  for (unsigned j = start; j < start + map.size(); j++) {
263  // There can be only one dynamic sized dimension among dimensions
264  // expanded from the i-th dimension in srcShape.
265  // For example, if srcDim = 8, then the expanded shape could be <2x?x2>,
266  // but not <2x?x?>.
267  if (staticDstShape[j] == ShapedType::kDynamic) {
268  // The expanded dimension has dynamic size. We compute the dimension
269  // by dividing srcDim by the product of the static dimensions.
270  int64_t product = 1;
271  for (unsigned k = start; k < start + map.size(); k++) {
272  if (staticDstShape[k] != ShapedType::kDynamic) {
273  product *= staticDstShape[k];
274  }
275  }
276  // Compute the dynamic dimension size.
277  Value productVal = constantIndex(rewriter, loc, product);
278  Value dynamicSize =
279  rewriter.create<arith::DivUIOp>(loc, srcDim, productVal);
280  dstShape.push_back(dynamicSize);
281  } else {
282  // The expanded dimension is statically known.
283  dstShape.push_back(constantIndex(rewriter, loc, staticDstShape[j]));
284  }
285  }
286  start = start + map.size();
287  }
288  assert(start == staticDstShape.size());
289 }

291 void mlir::sparse_tensor::translateIndicesArray(
292  OpBuilder &builder, Location loc,
293  ArrayRef<ReassociationIndices> reassociation, ValueRange srcIndices,
294  ArrayRef<Value> srcShape, ArrayRef<Value> dstShape,
295  SmallVectorImpl<Value> &dstIndices) {
296  unsigned i = 0;
297  unsigned start = 0;
298  unsigned dstRank = dstShape.size();
299  unsigned srcRank = srcShape.size();
300  assert(srcRank == srcIndices.size());
301  bool isCollapse = srcRank > dstRank;
302  ArrayRef<Value> shape = isCollapse ? srcShape : dstShape;
303  // Iterate over reassociation map.
304  for (const auto &map : llvm::enumerate(reassociation)) {
305  // Prepare strides information in dimension slice.
306  Value linear = constantIndex(builder, loc, 1);
307  for (unsigned j = start, end = start + map.value().size(); j < end; j++) {
308  linear = builder.create<arith::MulIOp>(loc, linear, shape[j]);
309  }
310  // Start expansion.
311  Value val;
312  if (!isCollapse)
313  val = srcIndices[i];
314  // Iterate over dimension slice.
315  for (unsigned j = start, end = start + map.value().size(); j < end; j++) {
316  linear = builder.create<arith::DivUIOp>(loc, linear, shape[j]);
317  if (isCollapse) {
318  Value old = srcIndices[j];
319  Value mul = builder.create<arith::MulIOp>(loc, old, linear);
320  val = val ? builder.create<arith::AddIOp>(loc, val, mul) : mul;
321  } else {
322  Value old = val;
323  val = builder.create<arith::DivUIOp>(loc, val, linear);
324  assert(dstIndices.size() == j);
325  dstIndices.push_back(val);
326  val = builder.create<arith::RemUIOp>(loc, old, linear);
327  }
328  }
329  // Finalize collapse.
330  if (isCollapse) {
331  assert(dstIndices.size() == i);
332  dstIndices.push_back(val);
333  }
334  start += map.value().size();
335  i++;
336  }
337  assert(dstIndices.size() == dstRank);
338 }

340 FlatSymbolRefAttr mlir::sparse_tensor::getFunc(ModuleOp module, StringRef name,
341  TypeRange resultType,
342  ValueRange operands,
343  EmitCInterface emitCInterface) {
344  MLIRContext *context = module.getContext();
345  auto result = SymbolRefAttr::get(context, name);
346  auto func = module.lookupSymbol<func::FuncOp>(result.getAttr());
347  if (!func) {
348  OpBuilder moduleBuilder(module.getBodyRegion());
349  func = moduleBuilder.create<func::FuncOp>(
350  module.getLoc(), name,
351  FunctionType::get(context, operands.getTypes(), resultType));
352  func.setPrivate();
353  if (static_cast<bool>(emitCInterface))
354  func->setAttr(LLVM::LLVMDialect::getEmitCWrapperAttrName(),
355  UnitAttr::get(context));
356  }
357  return result;
358 }

495 void sparse_tensor::foreachInSparseConstant(
496  Location loc, RewriterBase &rewriter, SparseElementsAttr attr,
497  function_ref<void(ArrayRef<Value>, Value)> callback) {
498  int64_t rank = attr.getType().getRank();
499  // Foreach on constant.
500  DenseElementsAttr indicesAttr = attr.getIndices();
501  DenseElementsAttr valuesAttr = attr.getValues();
502 
503  SmallVector<Value> coords;
504  for (int i = 0, e = valuesAttr.size(); i < e; i++) {
505  coords.clear();
506  for (int j = 0; j < rank; j++) {
507  auto coordAttr = indicesAttr.getValues<IntegerAttr>()[i * rank + j];
508  auto coord =
509  rewriter.create<arith::ConstantIndexOp>(loc, coordAttr.getInt());
510  // Remaps coordinates.
511  coords.push_back(coord);
512  }
513  Value val;
514  if (attr.getElementType().isa<ComplexType>()) {
515  auto valAttr = valuesAttr.getValues<ArrayAttr>()[i];
516  val = rewriter.create<complex::ConstantOp>(loc, attr.getElementType(),
517  valAttr);
518  } else {
519  auto valAttr = valuesAttr.getValues<TypedAttr>()[i];
520  // Remaps value.
521  val = rewriter.create<arith::ConstantOp>(loc, valAttr);
522  }
523  assert(val);
524  callback(coords, val);
525  }
