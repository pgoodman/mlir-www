24 void registerFromLLVMIRTranslation() {
25  TranslateToMLIRRegistration registration(
26  "import-llvm", "translate mlir to llvmir",
27  [](llvm::SourceMgr &sourceMgr,
28  MLIRContext *context) -> OwningOpRef<Operation *> {
29  llvm::SMDiagnostic err;
30  llvm::LLVMContext llvmContext;
31  std::unique_ptr<llvm::Module> llvmModule =
32  llvm::parseIR(*sourceMgr.getMemoryBuffer(sourceMgr.getMainFileID()),
33  err, llvmContext);
34  if (!llvmModule) {
35  std::string errStr;
36  llvm::raw_string_ostream errStream(errStr);
37  err.print(/*ProgName=*/"", errStream);
38  emitError(UnknownLoc::get(context)) << errStream.str();
39  return {};
40  }
41  return translateLLVMIRToModule(std::move(llvmModule), context);
42  });
43 }
