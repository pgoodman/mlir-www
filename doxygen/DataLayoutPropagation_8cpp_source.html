33 // The struct contains the infomation about mapping packing information to
34 // the iteration domain of Linalg ops.
35 struct PackInfo {
36  int64_t getNumTiledLoops() const { return tileToPointMapping.size(); };
37  // InnerDimsPos on iteration domain, which follows the order in pack ops.
38  SmallVector<int64_t> tiledDimsPos;
39  // The sizes of tiling data dimensions on iteration domain.
40  llvm::DenseMap<int64_t, OpFoldResult> domainDimAndTileMapping;
41  // The mapping from a dimension of iteration domain to the corresponding inner
42  // tiling dimension on iteration domain.
43  llvm::DenseMap<int64_t, int64_t> tileToPointMapping;
44  // The permutation of outer dims (on domain).
45  SmallVector<int64_t> outerDimsOnDomainPerm;
46  Optional<Value> paddingValue;
47 };

49 static PackInfo getPackingInfoFromConsumer(
50  AffineMap indexingMap, ArrayRef<OpFoldResult> innerTileSizes,
51  ArrayRef<int64_t> innerDimsPos, ArrayRef<int64_t> outerDimsPerm,
52  Optional<Value> paddingValue = std::nullopt) {
53  LLVM_DEBUG(
54  { llvm::dbgs() << "--- Construct PackInfo From A Consumer ---\n"; });
55  PackInfo packInfo;
56  packInfo.paddingValue = paddingValue;
57  int64_t origNumDims = indexingMap.getNumDims();
58  SmallVector<AffineExpr> exprs(indexingMap.getResults());
59  for (auto [index, innerDimPos, tileSize] :
60  llvm::zip_equal(llvm::seq<unsigned>(0, innerDimsPos.size()),
61  innerDimsPos, innerTileSizes)) {
62  int64_t domainDimPos =
63  exprs[innerDimPos].cast<AffineDimExpr>().getPosition();
64  packInfo.tiledDimsPos.push_back(domainDimPos);
65  packInfo.domainDimAndTileMapping[domainDimPos] = tileSize;
66  packInfo.tileToPointMapping[domainDimPos] = origNumDims + index;
67  LLVM_DEBUG({
68  llvm::dbgs() << "map innerDimPos=" << innerDimPos
69  << " to iteration dimension (d" << domainDimPos << ", d"
70  << packInfo.tileToPointMapping[domainDimPos]
71  << "), which has size=("
72  << packInfo.domainDimAndTileMapping[domainDimPos] << ")\n";
73  });
74  }
75 
76  for (auto dim : outerDimsPerm)
77  packInfo.outerDimsOnDomainPerm.push_back(indexingMap.getDimPosition(dim));
78  if (!packInfo.outerDimsOnDomainPerm.empty()) {
79  LLVM_DEBUG({
80  llvm::dbgs() << "map outer dimsDimsPerm to ";
81  for (auto dim : packInfo.outerDimsOnDomainPerm)
82  llvm::dbgs() << dim << " ";
83  llvm::dbgs() << "\n";
84  });
85  }
86 
87  return packInfo;
88 }

90 /// Returns a tuple for packed operand and indexing_map with the assumptions:
91 /// 1) The generic op is the producer of the pack op.
92 /// 2) The generic op has only one result.
93 /// If the operand is a scalar or packing dimensions are all irrelevant to the
94 /// operand, the opreand and the updated indexing map will be returned.
95 /// Otherwise, it returns the packed operand and the updated indexing map. E.g.,
96 ///
97 /// #map0 = affine_map<(d0, d1) -> (d0, d1)>
98 /// #map1 = affine_map<(d0, d1) -> (d0)>
99 /// #map2 = affine_map<(d0, d1) -> (d1)>
100 /// %0 = linalg.generic {indexing_maps = [#map1, #map2, #map0],
101 /// iterator_types = ["parallel", "parallel"]}
102 /// ins(%arg0, %arg1 : tensor<?xf32>, tensor<?xf32>)
103 /// outs(%init : tensor<?x?xf32>) {
104 /// ^bb0(%arg3: f32, %arg4: f32, %arg5: f32):
105 /// %4 = arith.addf %arg3, %arg4 : f32
106 /// linalg.yield %4 : f32
107 /// } -> tensor<?x?xf32>
108 /// %1 = tensor.pack %0
109 /// inner_dims_pos = [0, 1]
110 /// inner_tiles = [8, 2]
111 /// into %dest : tensor<?x?xf32> -> tensor<?x?x8x2xf32>
112 ///
113 /// Taking the first input operand as an example, the inner tile size of d1 is
114 /// 8. Thus, the below operation and `affine_map<(d0, d1, d2, d3)> ->
115 /// affine_map<(d1, d3)>` will be returned.
116 ///
117 /// %pack = tensor.pack %arg0
118 /// inner_dims_pos = [0]
119 /// inner_tiles = [8]
120 /// into %init : tensor<?xf32> -> tensor<?x8xf32>
121 static std::tuple<Value, AffineMap>
122 getOrCreatePackedViewOfOperand(OpBuilder &b, Location loc, PackInfo packInfo,
123  GenericOp genericOp, OpOperand *opOperand) {
124  int64_t numOrigLoops = genericOp.getNumLoops();
125  int64_t numInnerLoops = packInfo.getNumTiledLoops();
126  int64_t numLoops = numOrigLoops + numInnerLoops;
127  AffineMap origIndexingMap = genericOp.getMatchingIndexingMap(opOperand);
128  llvm::DenseMap<int64_t, int64_t> domainDimToOperandDim;
129  SmallVector<AffineExpr> exprs(origIndexingMap.getResults());
130  if (genericOp.isScalar(opOperand))
131  return std::make_tuple(opOperand->get(),
132  AffineMap::get(numLoops, 0, exprs, b.getContext()));
133 
134  // Step 1. Construct the information of packing data dimensions; append inner
135  // dimensions to the indexing maps for the operand.
136  for (auto [index, expr] : llvm::enumerate(exprs)) {
137  int64_t dimPos = expr.cast<AffineDimExpr>().getPosition();
138  domainDimToOperandDim[dimPos] = index;
139  }
140  SmallVector<int64_t> innerDimsPos;
141  SmallVector<OpFoldResult> innerTileSizes;
142  for (auto dimPos : packInfo.tiledDimsPos) {
143  if (!domainDimToOperandDim.count(dimPos))
144  continue;
145  int64_t index = domainDimToOperandDim[dimPos];
146  innerTileSizes.push_back(packInfo.domainDimAndTileMapping[dimPos]);
147  innerDimsPos.push_back(index);
148  exprs.push_back(b.getAffineDimExpr(packInfo.tileToPointMapping[dimPos]));
149  }
150 
151  // Step 2. Fold transpose variants (i.e., outerDimsPerm) into generic op.
152  // TODO: should we propagate the permutation of outer dims to the pack op?
153  SmallVector<int64_t> outerDimsPerm;
154  if (!packInfo.outerDimsOnDomainPerm.empty()) {
155  SmallVector<int64_t> inversedOuterPerm =
156  invertPermutationVector(packInfo.outerDimsOnDomainPerm);
157  for (auto i : llvm::seq<unsigned>(0, origIndexingMap.getNumResults())) {
158  int64_t dimPos = exprs[i].cast<AffineDimExpr>().getPosition();
159  exprs[i] = b.getAffineDimExpr(inversedOuterPerm[dimPos]);
160  }
161  }
162  auto indexingMap = AffineMap::get(numLoops, 0, exprs, b.getContext());
163 
164  // The operand does not have dimensions that relates to pack op.
165  if (innerDimsPos.empty())
166  return std::make_tuple(opOperand->get(), indexingMap);
167 
168  auto empty = tensor::PackOp::createDestinationTensor(
169  b, loc, opOperand->get(), innerTileSizes, innerDimsPos, outerDimsPerm);
170  auto packedOperand = b.create<tensor::PackOp>(
171  loc, opOperand->get(), empty, innerDimsPos, innerTileSizes,
172  packInfo.paddingValue, outerDimsPerm);
173  return std::make_tuple(packedOperand, indexingMap);
174 }

176 /// Bubbles up tensor.pack op through elementwise generic op. This
177 /// swap pack(generic) to generic(pack). The new generic op works on packed
178 /// domain; pack ops are created for input and output operands. E.g.,
179 ///
180 /// #map0 = affine_map<(d0, d1) -> (d0, d1)>
181 /// %0 = tensor.dim %arg0, %c0 : tensor<?x?xf32>
182 /// %1 = tensor.dim %arg0, %c1 : tensor<?x?xf32>
183 /// %2 = tensor.empty(%0, %1) : tensor<?x?xf32>
184 /// %3 = linalg.generic {indexing_maps = [#map0, #map0],
185 /// iterator_types = ["parallel", "parallel"]}
186 /// ins(%arg0 : tensor<?x?xf32>)
187 /// outs(%2 : tensor<?x?xf32>) {
188 /// ^bb0(%arg3: f32, %arg4: f32):
189 /// %4 = arith.addf %arg3, %arg3 : f32
190 /// linalg.yield %4 : f32
191 /// } -> tensor<?x?xf32>
192 /// %4 = tensor.pack %3
193 /// inner_dims_pos = [0, 1]
194 /// inner_tiles = [8, 2]
195 /// into %dest : tensor<?x?xf32> -> tensor<?x?x8x2xf32>
196 ///
197 /// will be converted to
198 ///
199 /// #map = affine_map<()[s0] -> (s0 ceildiv 8)>
200 /// #map1 = affine_map<()[s0] -> (s0 ceildiv 2)>
201 /// #map2 = affine_map<(d0, d1, d2, d3) -> (d0, d1, d2, d3)>
202 /// %dim = tensor.dim %arg0, %c0 : tensor<?x?xf32>
203 /// %dim_0 = tensor.dim %arg0, %c1 : tensor<?x?xf32>
204 /// %0 = affine.apply #map()[%dim]
205 /// %1 = affine.apply #map1()[%dim_0]
206 /// %2 = tensor.empty(%0, %1) : tensor<?x?x8x2xf32>
207 /// %pack = tensor.pack %arg0
208 /// inner_dims_pos = [0, 1]
209 /// inner_tiles = [8, 2]
210 /// into %2 : tensor<?x?xf32> -> tensor<?x?x8x2xf32>
211 /// %3 = linalg.generic {indexing_maps = [#map2, #map2],
212 /// iterator_types = ["parallel", "parallel", "parallel", "parallel"]}
213 /// ins(%pack : tensor<?x?x8x2xf32>)
214 /// outs(%arg1 : tensor<?x?x8x2xf32>) {
215 /// ^bb0(%in: f32, %out: f32):
216 /// %4 = arith.addf %in, %in : f32
217 /// linalg.yield %4 : f32
218 /// } -> tensor<?x?x8x2xf32>
219 static FailureOr<GenericOp>
220 bubbleUpPackOpThroughElemGenericOp(RewriterBase &rewriter,
221  tensor::PackOp packOp) {
222  auto genericOp = packOp.getSource().getDefiningOp<GenericOp>();
223  if (!genericOp)
224  return failure();
225 
226  if (!isElementwise(genericOp))
227  return failure();
228 
229  // TODO: Relax the restriction. We are able to bubble up the pack op through
230  // multi-result generic op. It just needs more work.
231  if (genericOp.getNumResults() != 1)
232  return failure();
233 
234  // TODO: Add an option for allowing padding values. It could introduce
235  // undefined behavior if we unconditionally propagate pack op through all
236  // the ops. E.g., if the padding value is zero and there are division ops in
237  // a generic op. Some values of padding area could be NaN (0/0).
238  if (packOp.getPaddingValue())
239  return failure();
240 
241  OpOperand *opOperand = genericOp.getDpsInitOperand(0);
242  auto packInfo = getPackingInfoFromConsumer(
243  genericOp.getMatchingIndexingMap(opOperand), packOp.getMixedTiles(),
244  packOp.getInnerDimsPos(), packOp.getOuterDimsPerm(),
245  packOp.getPaddingValue());
246 
247  Location loc = packOp.getLoc();
248  SmallVector<Value> inputOperands;
249  SmallVector<AffineMap> indexingMaps;
250  for (OpOperand *inputOperand : genericOp.getDpsInputOperands()) {
251  auto [packedOperand, packedIndexingMap] = getOrCreatePackedViewOfOperand(
252  rewriter, loc, packInfo, genericOp, inputOperand);
253  inputOperands.push_back(packedOperand);
254  indexingMaps.push_back(packedIndexingMap);
255  }
256 
257  int64_t numLoops = genericOp.getNumLoops();
258  int64_t numInnerLoops = packInfo.getNumTiledLoops();
259  int64_t newNumLoops = numLoops + numInnerLoops;
260  SmallVector<utils::IteratorType> iterTypes =
261  genericOp.getIteratorTypesArray();
262  iterTypes.append(numInnerLoops, utils::IteratorType::parallel);
263 
264  // Rebuild the indexing map for the corresponding init operand.
265  auto [packedOutOperand, packedOutIndexingMap] =
266  getOrCreatePackedViewOfOperand(rewriter, loc, packInfo, genericOp,
267  opOperand);
268  SmallVector<AffineExpr> outExprs(
269  packedOutIndexingMap.getResults().drop_back(numInnerLoops));
270  // Apply transpose to the indexing map, because we'll replace the init operand
271  // with the destination of pack op.
272  auto outerDimsPerm = packOp.getOuterDimsPerm();
273  if (!outerDimsPerm.empty()) {
274  applyPermutationToVector<AffineExpr>(outExprs, outerDimsPerm);
275  }
276  for (int i = 0; i < numInnerLoops; ++i)
277  outExprs.push_back(rewriter.getAffineDimExpr(numLoops + i));
278  AffineMap outMap =
279  AffineMap::get(newNumLoops, 0, outExprs, rewriter.getContext());
280  indexingMaps.push_back(outMap);
281 
282  auto newGenericOp = rewriter.create<linalg::GenericOp>(
283  loc, packOp.getDestType(), inputOperands, packOp.getDest(), indexingMaps,
284  iterTypes, /*bodyBuild=*/nullptr,
285  linalg::getPrunedAttributeList(genericOp));
286  rewriter.cloneRegionBefore(genericOp.getRegion(), newGenericOp.getRegion(),
287  newGenericOp.getRegion().begin());
288  return newGenericOp;
289 }

291 // Wrapper pattern that applies bubbleUpPackOpThroughElemGenericOp method.
292 struct BubbleUpPackOpThroughElemGenericOpPattern
293  : public OpRewritePattern<tensor::PackOp> {
294  using OpRewritePattern<tensor::PackOp>::OpRewritePattern;
295 
296  LogicalResult matchAndRewrite(tensor::PackOp packOp,
297  PatternRewriter &rewriter) const override {
298  auto genericOp = bubbleUpPackOpThroughElemGenericOp(rewriter, packOp);
299  if (failed(genericOp))
300  return failure();
301  rewriter.replaceOp(packOp, genericOp->getResults());
302  return success();
303  }
