25 void DebugImporter::translate(llvm::Function *func, LLVMFuncOp funcOp) {
26  if (!func->getSubprogram())
27  return;
28 
29  // Add a fused location to link the subprogram information.
30  StringAttr name = StringAttr::get(context, func->getSubprogram()->getName());
31  funcOp->setLoc(FusedLocWith<DISubprogramAttr>::get(
32  {NameLoc::get(name)}, translate(func->getSubprogram()), context));
33 }

53 DICompositeTypeAttr DebugImporter::translateImpl(llvm::DICompositeType *node) {
54  std::optional<DIFlags> flags = symbolizeDIFlags(node->getFlags());
55  SmallVector<DINodeAttr> elements;
56  for (llvm::DINode *element : node->getElements()) {
57  assert(element && "expected a non-null element type");
58  elements.push_back(translate(element));
59  }
60  return DICompositeTypeAttr::get(
61  context, node->getTag(), StringAttr::get(context, node->getName()),
62  translate(node->getFile()), node->getLine(), translate(node->getScope()),
63  translate(node->getBaseType()), flags.value_or(DIFlags::Zero),
64  node->getSizeInBits(), node->getAlignInBits(), elements);
65 }

117 DISubrangeAttr DebugImporter::translateImpl(llvm::DISubrange *node) {
118  auto getIntegerAttrOrNull = [&](llvm::DISubrange::BoundType data) {
119  if (auto *constInt = llvm::dyn_cast_or_null<llvm::ConstantInt *>(data))
120  return IntegerAttr::get(IntegerType::get(context, 64),
121  constInt->getSExtValue());
122  return IntegerAttr();
123  };
124  return DISubrangeAttr::get(context, getIntegerAttrOrNull(node->getCount()),
125  getIntegerAttrOrNull(node->getLowerBound()),
126  getIntegerAttrOrNull(node->getUpperBound()),
127  getIntegerAttrOrNull(node->getStride()));

131 DebugImporter::translateImpl(llvm::DISubroutineType *node) {
132  // Separate the result type since it is null for void functions.
133  DITypeAttr resultType = translate(*node->getTypeArray().begin());
134  SmallVector<DITypeAttr> argumentTypes;
135  for (llvm::DIType *type : llvm::drop_begin(node->getTypeArray())) {
136  assert(type && "expected a non-null argument type");
137  argumentTypes.push_back(translate(type));
138  }
139  return DISubroutineTypeAttr::get(context, node->getCC(), resultType,
140  argumentTypes);

147 DINodeAttr DebugImporter::translate(llvm::DINode *node) {
148  if (!node)
149  return nullptr;
150 
151  // Check for a cached instance.
152  if (DINodeAttr attr = nodeToAttr.lookup(node))
153  return attr;
154 
155  // Convert the debug metadata if possible.
156  auto translateNode = [this](llvm::DINode *node) -> DINodeAttr {
157  if (auto *casted = dyn_cast<llvm::DIBasicType>(node))
158  return translateImpl(casted);
159  if (auto *casted = dyn_cast<llvm::DICompileUnit>(node))
160  return translateImpl(casted);
161  if (auto *casted = dyn_cast<llvm::DICompositeType>(node))
162  return translateImpl(casted);
163  if (auto *casted = dyn_cast<llvm::DIDerivedType>(node))
164  return translateImpl(casted);
165  if (auto *casted = dyn_cast<llvm::DIFile>(node))
166  return translateImpl(casted);
167  if (auto *casted = dyn_cast<llvm::DILexicalBlock>(node))
168  return translateImpl(casted);
169  if (auto *casted = dyn_cast<llvm::DILexicalBlockFile>(node))
170  return translateImpl(casted);
171  if (auto *casted = dyn_cast<llvm::DILocalVariable>(node))
172  return translateImpl(casted);
173  if (auto *casted = dyn_cast<llvm::DISubprogram>(node))
174  return translateImpl(casted);
175  if (auto *casted = dyn_cast<llvm::DISubrange>(node))
176  return translateImpl(casted);
177  if (auto *casted = dyn_cast<llvm::DISubroutineType>(node))
178  return translateImpl(casted);
179  return nullptr;
180  };
181  if (DINodeAttr attr = translateNode(node)) {
182  nodeToAttr.insert({node, attr});
183  return attr;
184  }
185  return nullptr;
186 }

192 Location DebugImporter::translateLoc(llvm::DILocation *loc) {
193  if (!loc)
194  return UnknownLoc::get(context);
195 
196  // Get the file location of the instruction.
197  Location result = FileLineColLoc::get(context, loc->getFilename(),
198  loc->getLine(), loc->getColumn());
199 
200  // Add call site information, if available.
201  if (llvm::DILocation *inlinedAt = loc->getInlinedAt())
202  result = CallSiteLoc::get(result, translateLoc(inlinedAt));
203 
204  // Add scope information.
205  assert(loc->getScope() && "expected non-null scope");
206  result = FusedLocWith<DIScopeAttr>::get({result}, translate(loc->getScope()),
207  context);
208  return result;
209 }
