19 template <typename ReshapeOp>
20 struct FoldEmptyTensorWithReshapeOp : public OpRewritePattern<ReshapeOp> {
21  using OpRewritePattern<ReshapeOp>::OpRewritePattern;
22 
23  LogicalResult matchAndRewrite(ReshapeOp reshapeOp,
24  PatternRewriter &rewriter) const override {
25  if (!reshapeOp.getSrc().template getDefiningOp<EmptyOp>())
26  return failure();
27  Location loc = reshapeOp.getLoc();
28  ReifiedRankedShapedTypeDims resultShapes;
29  ReifyRankedShapedTypeOpInterface reifyShapedTypeInterface =
30  cast<ReifyRankedShapedTypeOpInterface>(reshapeOp.getOperation());
31  if (failed(reifyShapedTypeInterface.reifyResultShapes(rewriter,
32  resultShapes)) ||
33  !llvm::hasSingleElement(resultShapes))
34  return failure();
35  // TODO: Do not drop tensor type encoding.
36  Value emptyTensor =
37  rewriter.create<EmptyOp>(loc, getAsOpFoldResult(resultShapes[0]),
38  reshapeOp.getResultType().getElementType());
39  if (emptyTensor.getType() != reshapeOp.getResultType()) {
40  rewriter.replaceOpWithNewOp<tensor::CastOp>(
41  reshapeOp, reshapeOp.getResultType(), emptyTensor);
42  } else {
43  rewriter.replaceOp(reshapeOp, emptyTensor);
44  }
45  return success();
46  }
47 };

49 /// `tensor.empty` does not define any tensor contents, so a slice of a
50 /// `tensor.empty` can be canonicalized to a smaller `tensor.empty`.
51 struct FoldEmptyTensorWithExtractSliceOp
52  : public OpRewritePattern<ExtractSliceOp> {
53  using OpRewritePattern<ExtractSliceOp>::OpRewritePattern;
54 
55  LogicalResult matchAndRewrite(ExtractSliceOp sliceOp,
56  PatternRewriter &rewriter) const override {
57  if (!sliceOp.getSource().getDefiningOp<EmptyOp>())
58  return failure();
59 
60  // ExtractSliceOp may be rank-reducing; its dynamic sizes must be
61  // preserved as well as its result type.
62  auto tensorType = RankedTensorType::get(sliceOp.getType().getShape(),
63  sliceOp.getType().getElementType(),
64  sliceOp.getType().getEncoding());
65  rewriter.replaceOpWithNewOp<EmptyOp>(sliceOp, tensorType,
66  sliceOp.getSizes());
67  return success();
68  }
69 };
