34 Optional<SmallVector<int64_t>>
35 mlir::computeShapeRatio(ArrayRef<int64_t> shape, ArrayRef<int64_t> subShape) {
36  if (shape.size() < subShape.size())
37  return std::nullopt;
38  assert(llvm::all_of(shape, [](int64_t s) { return s > 0; }) &&
39  "shape must be nonnegative");
40  assert(llvm::all_of(subShape, [](int64_t s) { return s > 0; }) &&
41  "subShape must be nonnegative");
42 
43  // Starting from the end, compute the integer divisors.
44  std::vector<int64_t> result;
45  result.reserve(shape.size());
46  for (auto [size, subSize] :
47  llvm::zip(llvm::reverse(shape), llvm::reverse(subShape))) {
48  // If integral division does not occur, return and let the caller decide.
49  if (size % subSize != 0)
50  return std::nullopt;
51  result.push_back(size / subSize);
52  }
53  // At this point we computed the ratio (in reverse) for the common size.
54  // Fill with the remaining entries from the shape (still in reverse).
55  int commonSize = subShape.size();
56  std::copy(shape.rbegin() + commonSize, shape.rend(),
57  std::back_inserter(result));
58  // Reverse again to get it back in the proper order and return.
59  return SmallVector<int64_t>{result.rbegin(), result.rend()};
60 }
