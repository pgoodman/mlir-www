44 /// Compute and return the multi-dimensional integral ratio of `subShape` to
45 /// the trailing dimensions of `shape`. This represents how many times
46 /// `subShape` fits within `shape`.
47 /// If integral division is not possible, return std::nullopt.
48 /// The trailing `subShape.size()` entries of both shapes are assumed (and
49 /// enforced) to only contain noonnegative values.
50 ///
51 /// Examples:
52 /// - shapeRatio({3, 5, 8}, {2, 5, 2}) returns {3, 2, 1}.
53 /// - shapeRatio({3, 8}, {2, 5, 2}) returns std::nullopt (subshape has higher
54 /// rank).
55 /// - shapeRatio({42, 2, 10, 32}, {2, 5, 2}) returns {42, 1, 2, 16} which is
56 /// derived as {42(leading shape dim), 2/2, 10/5, 32/2}.
57 /// - shapeRatio({42, 2, 11, 32}, {2, 5, 2}) returns std::nullopt which is
58 /// derived as {42(leading shape dim), 2/2, 11/5(not divisible), 32/2}.

66 /// Apply the permutation defined by `permutation` to `inVec`.
67 /// Element `i` in `inVec` is mapped to location `j = permutation[i]`.
68 /// E.g.: for an input vector `inVec = ['a', 'b', 'c']` and a permutation vector
69 /// `permutation = [2, 0, 1]`, this function leaves `inVec = ['c', 'a', 'b']`.
70 template <typename T, unsigned N>
71 void applyPermutationToVector(SmallVector<T, N> &inVec,
72  ArrayRef<int64_t> permutation) {
73  SmallVector<T, N> auxVec(inVec.size());
74  for (const auto &en : enumerate(permutation))
75  auxVec[en.index()] = inVec[en.value()];
76  inVec = auxVec;
77 }
