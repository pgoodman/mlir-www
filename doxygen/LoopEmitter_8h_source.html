21 // SparseTensorLoopEmiter class, manages sparse tensors and helps to
22 // generate loop structure to (co)-iterate sparse tensors.
23 //
24 // An example usage:
25 // To generate the following loops over T1<?x?> and T2<?x?>
26 //
27 // for i in TENSOR_1_0 {
28 // for j : TENSOR_2_0 {
29 // for k : TENSOR_1_1 {}
30 // for k : TENSOR_2_1 {}
31 // }
32 // }
33 //
34 // One can use
35 //
36 // SparseTensorLoopEmiter loopEmiter({T1, T1});
37 // loopEmiter.initializeLoopEmit();
38 // loopEmiter.enterLoopOverTensorAtDim(T1, 0);
39 // loopEmiter.enterLoopOverTensorAtDim(T2, 0);
40 // loopEmiter.enterLoopOverTensorAtDim(T1, 1);
41 // loopEmiter.exitCurrentLoop();
42 // loopEmiter.enterLoopOverTensorAtDim(T2, 1);
43 // loopEmiter.exitCurrentLoop(); // exit k
44 // loopEmiter.exitCurrentLoop(); // exit j
45 // loopEmiter.exitCurrentLoop(); // exit i
46 //===----------------------------------------------------------------------===//

57  /// Takes an array of tensors inputs, on which the generated loops will
58  /// iterate on. The index of the tensor in the array is also the tensor id
59  /// (tid) used in related functions. If isSparseOut is set, loop emitter
60  /// assume that the sparse output tensor is empty, and will always generate
61  /// loops on it based on the dim sizes. An optional array could be provided
62  /// (by sparsification) to indicate the loop id sequence that will be
63  /// generated. It is used to establish the mapping between affineDimExpr to
64  /// the corresponding loop index in the loop stack that are maintained by the
65  /// loop emitter.

82  /// Enters a new loop sequence, the loops within the same sequence starts
83  /// from the break points of previous loop instead of starting over from 0.
84  /// e.g.,
85  /// {
86  /// // loop sequence start.
87  /// p0 = while(xxx)
88  /// ...
89  /// break p0
90  ///
91  /// // Starts loop from p0
92  /// for (i = p0; i < end; i++)
93  /// ...
94  /// // loop sequence end.
95  /// }
96  void enterNewLoopSeq(OpBuilder &builder, Location loc, ArrayRef<size_t> tids,
97  ArrayRef<size_t> dims);
98 
99  // exit the current loop sequence, this will reset universal index to 0.
100  void exitCurrentLoopSeq() {
101  assert(loopSeqStack.size() == loopStack.size() + 1);
102  loopSeqStack.pop_back();
103  }

142  /// Gets loop induction variable at the given level.

146  Value getLoopIV(size_t level) const {

210  /// Exits a for loop, returns the reduction results, e.g.,
211  /// For sequential for loops:
212  /// %ret = for () {
213  /// ...
214  /// %val = addi %args, %c
215  /// yield %val
216  /// }
217  /// For parallel loops, the following generated code by users:
218  /// %ret = parallel () init(%args) {
219  /// ...
220  /// %val = op %args, %c
221  /// }
222  /// will be transformed into
223  /// %ret = parallel () init(%args) {
224  /// ...
225  /// scf.reduce(%c) bb0(%0, %1){
226  /// %val = op %0, %1
227  /// scf.reduce.return %val
228  /// }
229  /// }
230  /// NOTE: only one instruction will be moved into reduce block,
231  /// transformation will fail if multiple instructions are used to compute
232  /// the reduction value. Return %ret to user, while %val is provided by
233  /// users (`reduc`).
234  void exitForLoop(RewriterBase &rewriter, Location loc,
235  MutableArrayRef<Value> reduc);
236 
237  /// Exits a while loop, returns the reduction results.
238  void exitCoIterationLoop(OpBuilder &builder, Location loc,
239  MutableArrayRef<Value> reduc);
