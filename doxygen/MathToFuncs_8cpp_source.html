82 template <typename Op>
83 LogicalResult
84 VecOpToScalarOp<Op>::matchAndRewrite(Op op, PatternRewriter &rewriter) const {
85  Type opType = op.getType();
86  Location loc = op.getLoc();
87  auto vecType = opType.template dyn_cast<VectorType>();
88 
89  if (!vecType)
90  return rewriter.notifyMatchFailure(op, "not a vector operation");
91  if (!vecType.hasRank())
92  return rewriter.notifyMatchFailure(op, "unknown vector rank");
93  ArrayRef<int64_t> shape = vecType.getShape();
94  int64_t numElements = vecType.getNumElements();
95 
96  Type resultElementType = vecType.getElementType();
97  Attribute initValueAttr;
98  if (resultElementType.isa<FloatType>())
99  initValueAttr = FloatAttr::get(resultElementType, 0.0);
100  else
101  initValueAttr = IntegerAttr::get(resultElementType, 0);
102  Value result = rewriter.create<arith::ConstantOp>(
103  loc, DenseElementsAttr::get(vecType, initValueAttr));
104  SmallVector<int64_t> strides = computeStrides(shape);
105  for (int64_t linearIndex = 0; linearIndex < numElements; ++linearIndex) {
106  SmallVector<int64_t> positions = delinearize(strides, linearIndex);
107  SmallVector<Value> operands;
108  for (Value input : op->getOperands())
109  operands.push_back(
110  rewriter.create<vector::ExtractOp>(loc, input, positions));
111  Value scalarOp =
112  rewriter.create<Op>(loc, vecType.getElementType(), operands);
113  result =
114  rewriter.create<vector::InsertOp>(loc, scalarOp, result, positions);
115  }
116  rewriter.replaceOp(op, result);
117  return success();
118 }

120 static FunctionType getElementalFuncTypeForOp(Operation *op) {
121  SmallVector<Type, 1> resultTys(op->getNumResults());
122  SmallVector<Type, 2> inputTys(op->getNumOperands());
123  std::transform(op->result_type_begin(), op->result_type_end(),
124  resultTys.begin(),
125  [](Type ty) { return getElementTypeOrSelf(ty); });
126  std::transform(op->operand_type_begin(), op->operand_type_end(),
127  inputTys.begin(),
128  [](Type ty) { return getElementTypeOrSelf(ty); });
129  return FunctionType::get(op->getContext(), inputTys, resultTys);
130 }

132 /// Create linkonce_odr function to implement the power function with
133 /// the given \p elementType type inside \p module. The \p elementType
134 /// must be IntegerType, an the created function has
135 /// 'IntegerType (*)(IntegerType, IntegerType)' function type.
136 ///
137 /// template <typename T>
138 /// T __mlir_math_ipowi_*(T b, T p) {
139 /// if (p == T(0))
140 /// return T(1);
141 /// if (p < T(0)) {
142 /// if (b == T(0))
143 /// return T(1) / T(0); // trigger div-by-zero
144 /// if (b == T(1))
145 /// return T(1);
146 /// if (b == T(-1)) {
147 /// if (p & T(1))
148 /// return T(-1);
149 /// return T(1);
150 /// }
151 /// return T(0);
152 /// }
153 /// T result = T(1);
154 /// while (true) {
155 /// if (p & T(1))
156 /// result *= b;
157 /// p >>= T(1);
158 /// if (p == T(0))
159 /// return result;
160 /// b *= b;
161 /// }
162 /// }
163 static func::FuncOp createElementIPowIFunc(ModuleOp *module, Type elementType) {
164  assert(elementType.isa<IntegerType>() &&
165  "non-integer element type for IPowIOp");
166 
167  ImplicitLocOpBuilder builder =
168  ImplicitLocOpBuilder::atBlockEnd(module->getLoc(), module->getBody());
169 
170  std::string funcName("__mlir_math_ipowi");
171  llvm::raw_string_ostream nameOS(funcName);
172  nameOS << '_' << elementType;
173 
174  FunctionType funcType = FunctionType::get(
175  builder.getContext(), {elementType, elementType}, elementType);
176  auto funcOp = builder.create<func::FuncOp>(funcName, funcType);
177  LLVM::linkage::Linkage inlineLinkage = LLVM::linkage::Linkage::LinkonceODR;
178  Attribute linkage =
179  LLVM::LinkageAttr::get(builder.getContext(), inlineLinkage);
180  funcOp->setAttr("llvm.linkage", linkage);
181  funcOp.setPrivate();
182 
183  Block *entryBlock = funcOp.addEntryBlock();
184  Region *funcBody = entryBlock->getParent();
185 
186  Value bArg = funcOp.getArgument(0);
187  Value pArg = funcOp.getArgument(1);
188  builder.setInsertionPointToEnd(entryBlock);
189  Value zeroValue = builder.create<arith::ConstantOp>(
190  elementType, builder.getIntegerAttr(elementType, 0));
191  Value oneValue = builder.create<arith::ConstantOp>(
192  elementType, builder.getIntegerAttr(elementType, 1));
193  Value minusOneValue = builder.create<arith::ConstantOp>(
194  elementType,
195  builder.getIntegerAttr(elementType,
196  APInt(elementType.getIntOrFloatBitWidth(), -1ULL,
197  /*isSigned=*/true)));
198 
199  // if (p == T(0))
200  // return T(1);
201  auto pIsZero =
202  builder.create<arith::CmpIOp>(arith::CmpIPredicate::eq, pArg, zeroValue);
203  Block *thenBlock = builder.createBlock(funcBody);
204  builder.create<func::ReturnOp>(oneValue);
205  Block *fallthroughBlock = builder.createBlock(funcBody);
206  // Set up conditional branch for (p == T(0)).
207  builder.setInsertionPointToEnd(pIsZero->getBlock());
208  builder.create<cf::CondBranchOp>(pIsZero, thenBlock, fallthroughBlock);
209 
210  // if (p < T(0)) {
211  builder.setInsertionPointToEnd(fallthroughBlock);
212  auto pIsNeg =
213  builder.create<arith::CmpIOp>(arith::CmpIPredicate::sle, pArg, zeroValue);
214  // if (b == T(0))
215  builder.createBlock(funcBody);
216  auto bIsZero =
217  builder.create<arith::CmpIOp>(arith::CmpIPredicate::eq, bArg, zeroValue);
218  // return T(1) / T(0);
219  thenBlock = builder.createBlock(funcBody);
220  builder.create<func::ReturnOp>(
221  builder.create<arith::DivSIOp>(oneValue, zeroValue).getResult());
222  fallthroughBlock = builder.createBlock(funcBody);
223  // Set up conditional branch for (b == T(0)).
224  builder.setInsertionPointToEnd(bIsZero->getBlock());
225  builder.create<cf::CondBranchOp>(bIsZero, thenBlock, fallthroughBlock);
226 
227  // if (b == T(1))
228  builder.setInsertionPointToEnd(fallthroughBlock);
229  auto bIsOne =
230  builder.create<arith::CmpIOp>(arith::CmpIPredicate::eq, bArg, oneValue);
231  // return T(1);
232  thenBlock = builder.createBlock(funcBody);
233  builder.create<func::ReturnOp>(oneValue);
234  fallthroughBlock = builder.createBlock(funcBody);
235  // Set up conditional branch for (b == T(1)).
236  builder.setInsertionPointToEnd(bIsOne->getBlock());
237  builder.create<cf::CondBranchOp>(bIsOne, thenBlock, fallthroughBlock);
238 
239  // if (b == T(-1)) {
240  builder.setInsertionPointToEnd(fallthroughBlock);
241  auto bIsMinusOne = builder.create<arith::CmpIOp>(arith::CmpIPredicate::eq,
242  bArg, minusOneValue);
243  // if (p & T(1))
244  builder.createBlock(funcBody);
245  auto pIsOdd = builder.create<arith::CmpIOp>(
246  arith::CmpIPredicate::ne, builder.create<arith::AndIOp>(pArg, oneValue),
247  zeroValue);
248  // return T(-1);
249  thenBlock = builder.createBlock(funcBody);
250  builder.create<func::ReturnOp>(minusOneValue);
251  fallthroughBlock = builder.createBlock(funcBody);
252  // Set up conditional branch for (p & T(1)).
253  builder.setInsertionPointToEnd(pIsOdd->getBlock());
254  builder.create<cf::CondBranchOp>(pIsOdd, thenBlock, fallthroughBlock);
255 
256  // return T(1);
257  // } // b == T(-1)
258  builder.setInsertionPointToEnd(fallthroughBlock);
259  builder.create<func::ReturnOp>(oneValue);
260  fallthroughBlock = builder.createBlock(funcBody);
261  // Set up conditional branch for (b == T(-1)).
262  builder.setInsertionPointToEnd(bIsMinusOne->getBlock());
263  builder.create<cf::CondBranchOp>(bIsMinusOne, pIsOdd->getBlock(),
264  fallthroughBlock);
265 
266  // return T(0);
267  // } // (p < T(0))
268  builder.setInsertionPointToEnd(fallthroughBlock);
269  builder.create<func::ReturnOp>(zeroValue);
270  Block *loopHeader = builder.createBlock(
271  funcBody, funcBody->end(), {elementType, elementType, elementType},
272  {builder.getLoc(), builder.getLoc(), builder.getLoc()});
273  // Set up conditional branch for (p < T(0)).
274  builder.setInsertionPointToEnd(pIsNeg->getBlock());
275  // Set initial values of 'result', 'b' and 'p' for the loop.
276  builder.create<cf::CondBranchOp>(pIsNeg, bIsZero->getBlock(), loopHeader,
277  ValueRange{oneValue, bArg, pArg});
278 
279  // T result = T(1);
280  // while (true) {
281  // if (p & T(1))
282  // result *= b;
283  // p >>= T(1);
284  // if (p == T(0))
285  // return result;
286  // b *= b;
287  // }
288  Value resultTmp = loopHeader->getArgument(0);
289  Value baseTmp = loopHeader->getArgument(1);
290  Value powerTmp = loopHeader->getArgument(2);
291  builder.setInsertionPointToEnd(loopHeader);
292 
293  // if (p & T(1))
294  auto powerTmpIsOdd = builder.create<arith::CmpIOp>(
295  arith::CmpIPredicate::ne,
296  builder.create<arith::AndIOp>(powerTmp, oneValue), zeroValue);
297  thenBlock = builder.createBlock(funcBody);
298  // result *= b;
299  Value newResultTmp = builder.create<arith::MulIOp>(resultTmp, baseTmp);
300  fallthroughBlock = builder.createBlock(funcBody, funcBody->end(), elementType,
301  builder.getLoc());
302  builder.setInsertionPointToEnd(thenBlock);
303  builder.create<cf::BranchOp>(newResultTmp, fallthroughBlock);
304  // Set up conditional branch for (p & T(1)).
305  builder.setInsertionPointToEnd(powerTmpIsOdd->getBlock());
306  builder.create<cf::CondBranchOp>(powerTmpIsOdd, thenBlock, fallthroughBlock,
307  resultTmp);
308  // Merged 'result'.
309  newResultTmp = fallthroughBlock->getArgument(0);
310 
311  // p >>= T(1);
312  builder.setInsertionPointToEnd(fallthroughBlock);
313  Value newPowerTmp = builder.create<arith::ShRUIOp>(powerTmp, oneValue);
314 
315  // if (p == T(0))
316  auto newPowerIsZero = builder.create<arith::CmpIOp>(arith::CmpIPredicate::eq,
317  newPowerTmp, zeroValue);
318  // return result;
319  thenBlock = builder.createBlock(funcBody);
320  builder.create<func::ReturnOp>(newResultTmp);
321  fallthroughBlock = builder.createBlock(funcBody);
322  // Set up conditional branch for (p == T(0)).
323  builder.setInsertionPointToEnd(newPowerIsZero->getBlock());
324  builder.create<cf::CondBranchOp>(newPowerIsZero, thenBlock, fallthroughBlock);
325 
326  // b *= b;
327  // }
328  builder.setInsertionPointToEnd(fallthroughBlock);
329  Value newBaseTmp = builder.create<arith::MulIOp>(baseTmp, baseTmp);
330  // Pass new values for 'result', 'b' and 'p' to the loop header.
331  builder.create<cf::BranchOp>(
332  ValueRange{newResultTmp, newBaseTmp, newPowerTmp}, loopHeader);
333  return funcOp;
334 }

336 /// Convert IPowI into a call to a local function implementing
337 /// the power operation. The local function computes a scalar result,
338 /// so vector forms of IPowI are linearized.
339 LogicalResult
340 IPowIOpLowering::matchAndRewrite(math::IPowIOp op,
341  PatternRewriter &rewriter) const {
342  auto baseType = op.getOperands()[0].getType().dyn_cast<IntegerType>();
343 
344  if (!baseType)
345  return rewriter.notifyMatchFailure(op, "non-integer base operand");
346 
347  // The outlined software implementation must have been already
348  // generated.
349  func::FuncOp elementFunc = getFuncOpCallback(baseType);
350  if (!elementFunc)
351  return rewriter.notifyMatchFailure(op, "missing software implementation");
352 
353  rewriter.replaceOpWithNewOp<func::CallOp>(op, elementFunc, op.getOperands());
354  return success();
355 }

357 /// Create linkonce_odr function to implement the power function with
358 /// the given \p funcType type inside \p module. The \p funcType must be
359 /// 'FloatType (*)(FloatType, IntegerType)' function type.
360 ///
361 /// template <typename T>
362 /// Tb __mlir_math_fpowi_*(Tb b, Tp p) {
363 /// if (p == Tp{0})
364 /// return Tb{1};
365 /// bool isNegativePower{p < Tp{0}}
366 /// bool isMin{p == std::numeric_limits<Tp>::min()};
367 /// if (isMin) {
368 /// p = std::numeric_limits<Tp>::max();
369 /// } else if (isNegativePower) {
370 /// p = -p;
371 /// }
372 /// Tb result = Tb{1};
373 /// Tb origBase = Tb{b};
374 /// while (true) {
375 /// if (p & Tp{1})
376 /// result *= b;
377 /// p >>= Tp{1};
378 /// if (p == Tp{0})
379 /// break;
380 /// b *= b;
381 /// }
382 /// if (isMin) {
383 /// result *= origBase;
384 /// }
385 /// if (isNegativePower) {
386 /// result = Tb{1} / result;
387 /// }
388 /// return result;
389 /// }
390 static func::FuncOp createElementFPowIFunc(ModuleOp *module,
391  FunctionType funcType) {
392  auto baseType = funcType.getInput(0).cast<FloatType>();
393  auto powType = funcType.getInput(1).cast<IntegerType>();
394  ImplicitLocOpBuilder builder =
395  ImplicitLocOpBuilder::atBlockEnd(module->getLoc(), module->getBody());
396 
397  std::string funcName("__mlir_math_fpowi");
398  llvm::raw_string_ostream nameOS(funcName);
399  nameOS << '_' << baseType;
400  nameOS << '_' << powType;
401  auto funcOp = builder.create<func::FuncOp>(funcName, funcType);
402  LLVM::linkage::Linkage inlineLinkage = LLVM::linkage::Linkage::LinkonceODR;
403  Attribute linkage =
404  LLVM::LinkageAttr::get(builder.getContext(), inlineLinkage);
405  funcOp->setAttr("llvm.linkage", linkage);
406  funcOp.setPrivate();
407 
408  Block *entryBlock = funcOp.addEntryBlock();
409  Region *funcBody = entryBlock->getParent();
410 
411  Value bArg = funcOp.getArgument(0);
412  Value pArg = funcOp.getArgument(1);
413  builder.setInsertionPointToEnd(entryBlock);
414  Value oneBValue = builder.create<arith::ConstantOp>(
415  baseType, builder.getFloatAttr(baseType, 1.0));
416  Value zeroPValue = builder.create<arith::ConstantOp>(
417  powType, builder.getIntegerAttr(powType, 0));
418  Value onePValue = builder.create<arith::ConstantOp>(
419  powType, builder.getIntegerAttr(powType, 1));
420  Value minPValue = builder.create<arith::ConstantOp>(
421  powType, builder.getIntegerAttr(powType, llvm::APInt::getSignedMinValue(
422  powType.getWidth())));
423  Value maxPValue = builder.create<arith::ConstantOp>(
424  powType, builder.getIntegerAttr(powType, llvm::APInt::getSignedMaxValue(
425  powType.getWidth())));
426 
427  // if (p == Tp{0})
428  // return Tb{1};
429  auto pIsZero =
430  builder.create<arith::CmpIOp>(arith::CmpIPredicate::eq, pArg, zeroPValue);
431  Block *thenBlock = builder.createBlock(funcBody);
432  builder.create<func::ReturnOp>(oneBValue);
433  Block *fallthroughBlock = builder.createBlock(funcBody);
434  // Set up conditional branch for (p == Tp{0}).
435  builder.setInsertionPointToEnd(pIsZero->getBlock());
436  builder.create<cf::CondBranchOp>(pIsZero, thenBlock, fallthroughBlock);
437 
438  builder.setInsertionPointToEnd(fallthroughBlock);
439  // bool isNegativePower{p < Tp{0}}
440  auto pIsNeg = builder.create<arith::CmpIOp>(arith::CmpIPredicate::sle, pArg,
441  zeroPValue);
442  // bool isMin{p == std::numeric_limits<Tp>::min()};
443  auto pIsMin =
444  builder.create<arith::CmpIOp>(arith::CmpIPredicate::eq, pArg, minPValue);
445 
446  // if (isMin) {
447  // p = std::numeric_limits<Tp>::max();
448  // } else if (isNegativePower) {
449  // p = -p;
450  // }
451  Value negP = builder.create<arith::SubIOp>(zeroPValue, pArg);
452  auto pInit = builder.create<arith::SelectOp>(pIsNeg, negP, pArg);
453  pInit = builder.create<arith::SelectOp>(pIsMin, maxPValue, pInit);
454 
455  // Tb result = Tb{1};
456  // Tb origBase = Tb{b};
457  // while (true) {
458  // if (p & Tp{1})
459  // result *= b;
460  // p >>= Tp{1};
461  // if (p == Tp{0})
462  // break;
463  // b *= b;
464  // }
465  Block *loopHeader = builder.createBlock(
466  funcBody, funcBody->end(), {baseType, baseType, powType},
467  {builder.getLoc(), builder.getLoc(), builder.getLoc()});
468  // Set initial values of 'result', 'b' and 'p' for the loop.
469  builder.setInsertionPointToEnd(pInit->getBlock());
470  builder.create<cf::BranchOp>(loopHeader, ValueRange{oneBValue, bArg, pInit});
471 
472  // Create loop body.
473  Value resultTmp = loopHeader->getArgument(0);
474  Value baseTmp = loopHeader->getArgument(1);
475  Value powerTmp = loopHeader->getArgument(2);
476  builder.setInsertionPointToEnd(loopHeader);
477 
478  // if (p & Tp{1})
479  auto powerTmpIsOdd = builder.create<arith::CmpIOp>(
480  arith::CmpIPredicate::ne,
481  builder.create<arith::AndIOp>(powerTmp, onePValue), zeroPValue);
482  thenBlock = builder.createBlock(funcBody);
483  // result *= b;
484  Value newResultTmp = builder.create<arith::MulFOp>(resultTmp, baseTmp);
485  fallthroughBlock = builder.createBlock(funcBody, funcBody->end(), baseType,
486  builder.getLoc());
487  builder.setInsertionPointToEnd(thenBlock);
488  builder.create<cf::BranchOp>(newResultTmp, fallthroughBlock);
489  // Set up conditional branch for (p & Tp{1}).
490  builder.setInsertionPointToEnd(powerTmpIsOdd->getBlock());
491  builder.create<cf::CondBranchOp>(powerTmpIsOdd, thenBlock, fallthroughBlock,
492  resultTmp);
493  // Merged 'result'.
494  newResultTmp = fallthroughBlock->getArgument(0);
495 
496  // p >>= Tp{1};
497  builder.setInsertionPointToEnd(fallthroughBlock);
498  Value newPowerTmp = builder.create<arith::ShRUIOp>(powerTmp, onePValue);
499 
500  // if (p == Tp{0})
501  auto newPowerIsZero = builder.create<arith::CmpIOp>(arith::CmpIPredicate::eq,
502  newPowerTmp, zeroPValue);
503  // break;
504  //
505  // The conditional branch is finalized below with a jump to
506  // the loop exit block.
507  fallthroughBlock = builder.createBlock(funcBody);
508 
509  // b *= b;
510  // }
511  builder.setInsertionPointToEnd(fallthroughBlock);
512  Value newBaseTmp = builder.create<arith::MulFOp>(baseTmp, baseTmp);
513  // Pass new values for 'result', 'b' and 'p' to the loop header.
514  builder.create<cf::BranchOp>(
515  ValueRange{newResultTmp, newBaseTmp, newPowerTmp}, loopHeader);
516 
517  // Set up conditional branch for early loop exit:
518  // if (p == Tp{0})
519  // break;
520  Block *loopExit = builder.createBlock(funcBody, funcBody->end(), baseType,
521  builder.getLoc());
522  builder.setInsertionPointToEnd(newPowerIsZero->getBlock());
523  builder.create<cf::CondBranchOp>(newPowerIsZero, loopExit, newResultTmp,
524  fallthroughBlock, ValueRange{});
525 
526  // if (isMin) {
527  // result *= origBase;
528  // }
529  newResultTmp = loopExit->getArgument(0);
530  thenBlock = builder.createBlock(funcBody);
531  fallthroughBlock = builder.createBlock(funcBody, funcBody->end(), baseType,
532  builder.getLoc());
533  builder.setInsertionPointToEnd(loopExit);
534  builder.create<cf::CondBranchOp>(pIsMin, thenBlock, fallthroughBlock,
535  newResultTmp);
536  builder.setInsertionPointToEnd(thenBlock);
537  newResultTmp = builder.create<arith::MulFOp>(newResultTmp, bArg);
538  builder.create<cf::BranchOp>(newResultTmp, fallthroughBlock);
539 
540  /// if (isNegativePower) {
541  /// result = Tb{1} / result;
542  /// }
543  newResultTmp = fallthroughBlock->getArgument(0);
544  thenBlock = builder.createBlock(funcBody);
545  Block *returnBlock = builder.createBlock(funcBody, funcBody->end(), baseType,
546  builder.getLoc());
547  builder.setInsertionPointToEnd(fallthroughBlock);
548  builder.create<cf::CondBranchOp>(pIsNeg, thenBlock, returnBlock,

564 LogicalResult
565 FPowIOpLowering::matchAndRewrite(math::FPowIOp op,
566  PatternRewriter &rewriter) const {

614 void ConvertMathToFuncsPass::preprocessPowOperations() {
615  ModuleOp module = getOperation();
616 
617  module.walk([&](Operation *op) {
618  TypeSwitch<Operation *>(op)
619  .Case<math::IPowIOp>([&](math::IPowIOp op) {
620  Type resultType = getElementTypeOrSelf(op.getResult().getType());
621 
622  // Generate the software implementation of this operation,
623  // if it has not been generated yet.
624  auto entry = powerFuncs.try_emplace(resultType, func::FuncOp{});
625  if (entry.second)
626  entry.first->second = createElementIPowIFunc(&module, resultType);
627  })
628  .Case<math::FPowIOp>([&](math::FPowIOp op) {
629  if (!isFPowIConvertible(op))
630  return;
631 
632  FunctionType funcType = getElementalFuncTypeForOp(op);
633 
634  // Generate the software implementation of this operation,
635  // if it has not been generated yet.
636  // FPowI implementations are mapped via the FunctionType
637  // created from the operation's result and operands.
638  auto entry = powerFuncs.try_emplace(funcType, func::FuncOp{});
639  if (entry.second)
640  entry.first->second = createElementFPowIFunc(&module, funcType);
641  });
642  });
643 }

645 void ConvertMathToFuncsPass::runOnOperation() {
646  ModuleOp module = getOperation();
647 
648  // Create outlined implementations for power operations.
649  preprocessPowOperations();
650 
651  RewritePatternSet patterns(&getContext());
652  patterns.add<VecOpToScalarOp<math::IPowIOp>, VecOpToScalarOp<math::FPowIOp>>(
653  patterns.getContext());
654 
655  // For the given Type Returns FuncOp stored in powerFuncs map.
656  auto getPowerFuncOpByType = [&](Type type) -> func::FuncOp {
657  auto it = powerFuncs.find(type);
658  if (it == powerFuncs.end())
659  return {};
660 
661  return it->second;
662  };
663  patterns.add<IPowIOpLowering, FPowIOpLowering>(patterns.getContext(),
664  getPowerFuncOpByType);
665 
666  ConversionTarget target(getContext());
667  target.addLegalDialect<arith::ArithDialect, cf::ControlFlowDialect,
668  func::FuncDialect, vector::VectorDialect>();
669  target.addIllegalOp<math::IPowIOp>();
670  target.addDynamicallyLegalOp<math::FPowIOp>(
671  [this](math::FPowIOp op) { return !isFPowIConvertible(op); });
672  if (failed(applyPartialConversion(module, target, std::move(patterns))))
673  signalPassFailure();
