59 /// Creates an attribute containing ABI and preferred alignment numbers parsed
60 /// a string. The string may be either "abi:preferred" or just "abi". In the
61 /// latter case, the prefrred alignment is considered equal to ABI alignment.
62 static DenseIntElementsAttr parseDataLayoutAlignment(MLIRContext &ctx,
63  StringRef spec) {
64  auto i32 = IntegerType::get(&ctx, 32);
65 
66  StringRef abiString, preferredString;
67  std::tie(abiString, preferredString) = spec.split(':');
68  int abi, preferred;
69  if (abiString.getAsInteger(/*Radix=*/10, abi))
70  return nullptr;
71 
72  if (preferredString.empty())
73  preferred = abi;
74  else if (preferredString.getAsInteger(/*Radix=*/10, preferred))
75  return nullptr;
76 
77  return DenseIntElementsAttr::get(VectorType::get({2}, i32), {abi, preferred});
78 }

239 /// Translate the given LLVM data layout into an MLIR equivalent using the DLTI
240 /// dialect.
241 DataLayoutSpecInterface
242 mlir::translateDataLayout(const llvm::DataLayout &dataLayout,
243  MLIRContext *context) {
244  assert(context && "expected MLIR context");
245  std::string layoutstr = dataLayout.getStringRepresentation();
246 
247  // Remaining unhandled default layout defaults
248  // e (little endian if not set)
249  // p[n]:64:64:64 (non zero address spaces have 64-bit properties)
250  std::string append =
251  "p:64:64:64-S0-i1:8:8-i8:8:8-i16:16:16-i32:32:32-i64:32:64-f16:16:16-f64:"
252  "64:64-f128:128:128-v64:64:64-v128:128:128-a:0:64";
253  if (layoutstr.empty())
254  layoutstr = append;
255  else
256  layoutstr = layoutstr + "-" + append;
257 
258  StringRef layout(layoutstr);
259 
260  SmallVector<DataLayoutEntryInterface> entries;
261  StringSet<> seen;
262  while (!layout.empty()) {
263  // Split at '-'.
264  std::pair<StringRef, StringRef> split = layout.split('-');
265  StringRef current;
266  std::tie(current, layout) = split;
267 
268  // Split at ':'.
269  StringRef kind, spec;
270  std::tie(kind, spec) = current.split(':');
271  if (seen.contains(kind))
272  continue;
273  seen.insert(kind);
274 
275  char symbol = kind.front();
276  StringRef parameter = kind.substr(1);
277 
278  if (symbol == 'i' || symbol == 'f') {
279  unsigned bitwidth;
280  if (parameter.getAsInteger(/*Radix=*/10, bitwidth))
281  return nullptr;
282  DenseIntElementsAttr params = parseDataLayoutAlignment(*context, spec);
283  if (!params)
284  return nullptr;
285  auto entry = DataLayoutEntryAttr::get(
286  symbol == 'i' ? static_cast<Type>(IntegerType::get(context, bitwidth))
287  : getDLFloatType(*context, bitwidth),
288  params);
289  entries.emplace_back(entry);
290  } else if (symbol == 'e' || symbol == 'E') {
291  auto value = StringAttr::get(
292  context, symbol == 'e' ? DLTIDialect::kDataLayoutEndiannessLittle
293  : DLTIDialect::kDataLayoutEndiannessBig);
294  auto entry = DataLayoutEntryAttr::get(
295  StringAttr::get(context, DLTIDialect::kDataLayoutEndiannessKey),
296  value);
297  entries.emplace_back(entry);
298  }
299  }
300 
301  return DataLayoutSpecAttr::get(context, entries);
302 }

342 void ModuleImport::setFastmathFlagsAttr(llvm::Instruction *inst,
343  Operation *op) const {
344  auto iface = cast<FastmathFlagsInterface>(op);
345 
346  // Even if the imported operation implements the fastmath interface, the
347  // original instruction may not have fastmath flags set. Exit if an
348  // instruction, such as a non floating-point function call, does not have
349  // fastmath flags.
350  if (!isa<llvm::FPMathOperator>(inst))
351  return;
352  llvm::FastMathFlags flags = inst->getFastMathFlags();
353 
354  // Set the fastmath bits flag-by-flag.
355  FastmathFlags value = {};
356  value = bitEnumSet(value, FastmathFlags::nnan, flags.noNaNs());
357  value = bitEnumSet(value, FastmathFlags::ninf, flags.noInfs());
358  value = bitEnumSet(value, FastmathFlags::nsz, flags.noSignedZeros());
359  value = bitEnumSet(value, FastmathFlags::arcp, flags.allowReciprocal());
360  value = bitEnumSet(value, FastmathFlags::contract, flags.allowContract());
361  value = bitEnumSet(value, FastmathFlags::afn, flags.approxFunc());
362  value = bitEnumSet(value, FastmathFlags::reassoc, flags.allowReassoc());
363  FastmathFlagsAttr attr = FastmathFlagsAttr::get(builder.getContext(), value);
364  iface->setAttr(iface.getFastmathAttrName(), attr);
365 }

367 // We only need integers, floats, doubles, and vectors and tensors thereof for
368 // attributes. Scalar and vector types are converted to the standard
369 // equivalents. Array types are converted to ranked tensors; nested array types
370 // are converted to multi-dimensional tensors or vectors, depending on the
371 // innermost type being a scalar or a vector.
372 Type ModuleImport::getStdTypeForAttr(Type type) {
373  if (!type)
374  return nullptr;
375 
376  if (type.isa<IntegerType, FloatType>())
377  return type;
378 
379  // LLVM vectors can only contain scalars.
380  if (LLVM::isCompatibleVectorType(type)) {
381  llvm::ElementCount numElements = LLVM::getVectorNumElements(type);
382  if (numElements.isScalable()) {
383  emitError(UnknownLoc::get(context)) << "scalable vectors not supported";
384  return nullptr;
385  }
386  Type elementType = getStdTypeForAttr(LLVM::getVectorElementType(type));
387  if (!elementType)
388  return nullptr;
389  return VectorType::get(numElements.getKnownMinValue(), elementType);
390  }
391 
392  // LLVM arrays can contain other arrays or vectors.
393  if (auto arrayType = type.dyn_cast<LLVMArrayType>()) {
394  // Recover the nested array shape.
395  SmallVector<int64_t, 4> shape;
396  shape.push_back(arrayType.getNumElements());
397  while (arrayType.getElementType().isa<LLVMArrayType>()) {
398  arrayType = arrayType.getElementType().cast<LLVMArrayType>();
399  shape.push_back(arrayType.getNumElements());
400  }
401 
402  // If the innermost type is a vector, use the multi-dimensional vector as
403  // attribute type.
404  if (LLVM::isCompatibleVectorType(arrayType.getElementType())) {
405  llvm::ElementCount numElements =
406  LLVM::getVectorNumElements(arrayType.getElementType());
407  if (numElements.isScalable()) {
408  emitError(UnknownLoc::get(context)) << "scalable vectors not supported";
409  return nullptr;
410  }
411  shape.push_back(numElements.getKnownMinValue());
412 
413  Type elementType = getStdTypeForAttr(
414  LLVM::getVectorElementType(arrayType.getElementType()));
415  if (!elementType)
416  return nullptr;
417  return VectorType::get(shape, elementType);
418  }
419 
420  // Otherwise use a tensor.
421  Type elementType = getStdTypeForAttr(arrayType.getElementType());
422  if (!elementType)
423  return nullptr;
424  return RankedTensorType::get(shape, elementType);
425  }
426 
427  return nullptr;
428 }

430 // Get the given constant as an attribute. Not all constants can be represented
431 // as attributes.
432 Attribute ModuleImport::getConstantAsAttr(llvm::Constant *value) {
433  if (auto *ci = dyn_cast<llvm::ConstantInt>(value))
434  return builder.getIntegerAttr(
435  IntegerType::get(context, ci->getType()->getBitWidth()),
436  ci->getValue());
437  if (auto *c = dyn_cast<llvm::ConstantDataArray>(value))
438  if (c->isString())
439  return builder.getStringAttr(c->getAsString());
440  if (auto *c = dyn_cast<llvm::ConstantFP>(value)) {
441  llvm::Type *type = c->getType();
442  FloatType floatTy;
443  if (type->isBFloatTy())
444  floatTy = FloatType::getBF16(context);
445  else
446  floatTy = getDLFloatType(*context, type->getScalarSizeInBits());
447  assert(floatTy && "unsupported floating point type");
448  return builder.getFloatAttr(floatTy, c->getValueAPF());
449  }
450  if (auto *f = dyn_cast<llvm::Function>(value))
451  return SymbolRefAttr::get(builder.getContext(), f->getName());
452 
453  // Convert constant data to a dense elements attribute.
454  if (auto *cd = dyn_cast<llvm::ConstantDataSequential>(value)) {
455  Type type = convertType(cd->getElementType());
456  auto attrType = getStdTypeForAttr(convertType(cd->getType()))
457  .dyn_cast_or_null<ShapedType>();
458  if (!attrType)
459  return nullptr;
460 
461  if (type.isa<IntegerType>()) {
462  SmallVector<APInt, 8> values;
463  values.reserve(cd->getNumElements());
464  for (unsigned i = 0, e = cd->getNumElements(); i < e; ++i)
465  values.push_back(cd->getElementAsAPInt(i));
466  return DenseElementsAttr::get(attrType, values);
467  }
468 
469  if (type.isa<Float32Type, Float64Type>()) {
470  SmallVector<APFloat, 8> values;
471  values.reserve(cd->getNumElements());
472  for (unsigned i = 0, e = cd->getNumElements(); i < e; ++i)
473  values.push_back(cd->getElementAsAPFloat(i));
474  return DenseElementsAttr::get(attrType, values);
475  }
476 
477  return nullptr;
478  }
479 
480  // Unpack constant aggregates to create dense elements attribute whenever
481  // possible. Return nullptr (failure) otherwise.
482  if (isa<llvm::ConstantAggregate>(value)) {
483  auto outerType = getStdTypeForAttr(convertType(value->getType()))
484  .dyn_cast_or_null<ShapedType>();
485  if (!outerType)
486  return nullptr;
487 
488  SmallVector<Attribute, 8> values;
489  SmallVector<int64_t, 8> shape;
490 
491  for (unsigned i = 0, e = value->getNumOperands(); i < e; ++i) {
492  auto nested = getConstantAsAttr(value->getAggregateElement(i))
493  .dyn_cast_or_null<DenseElementsAttr>();
494  if (!nested)
495  return nullptr;
496 
497  values.append(nested.value_begin<Attribute>(),
498  nested.value_end<Attribute>());
499  }
500 
501  return DenseElementsAttr::get(outerType, values);
502  }
503 
504  return nullptr;
505 }

507 GlobalOp ModuleImport::processGlobal(llvm::GlobalVariable *globalVar) {
508  if (globals.count(globalVar))
509  return globals[globalVar];
510 
511  // Insert the global after the last one or at the start of the module.
512  OpBuilder::InsertionGuard guard(builder);
513  if (!globalInsertionOp) {
514  builder.setInsertionPointToStart(mlirModule.getBody());
515  } else {
516  builder.setInsertionPointAfter(globalInsertionOp);
517  }
518 
519  Attribute valueAttr;
520  if (globalVar->hasInitializer())
521  valueAttr = getConstantAsAttr(globalVar->getInitializer());
522  Type type = convertType(globalVar->getValueType());
523 
524  uint64_t alignment = 0;
525  llvm::MaybeAlign maybeAlign = globalVar->getAlign();
526  if (maybeAlign.has_value()) {
527  llvm::Align align = *maybeAlign;
528  alignment = align.value();
529  }
530 
531  GlobalOp globalOp = builder.create<GlobalOp>(
532  UnknownLoc::get(context), type, globalVar->isConstant(),
533  convertLinkageFromLLVM(globalVar->getLinkage()), globalVar->getName(),
534  valueAttr, alignment, /*addr_space=*/globalVar->getAddressSpace(),
535  /*dso_local=*/globalVar->isDSOLocal(),
536  /*thread_local=*/globalVar->isThreadLocal());
537  globalInsertionOp = globalOp;
538 
539  if (globalVar->hasInitializer() && !valueAttr) {
540  clearBlockAndValueMapping();
541  Block *block = builder.createBlock(&globalOp.getInitializerRegion());
542  setConstantInsertionPointToStart(block);
543  FailureOr<Value> initializer =
544  convertConstantExpr(globalVar->getInitializer());
545  if (failed(initializer))
546  return {};
547  builder.create<ReturnOp>(globalOp.getLoc(), *initializer);
548  }
549  if (globalVar->hasAtLeastLocalUnnamedAddr()) {
550  globalOp.setUnnamedAddr(
551  convertUnnamedAddrFromLLVM(globalVar->getUnnamedAddr()));
552  }
553  if (globalVar->hasSection())
554  globalOp.setSection(globalVar->getSection());
555 
556  return globals[globalVar] = globalOp;
557 }

559 SetVector<llvm::Constant *>
560 ModuleImport::getConstantsToConvert(llvm::Constant *constant) {
561  // Traverse the constant dependencies in post order.
562  SmallVector<llvm::Constant *> workList;
563  SmallVector<llvm::Constant *> orderedList;
564  workList.push_back(constant);
565  while (!workList.empty()) {
566  llvm::Constant *current = workList.pop_back_val();
567  // Skip constants that have been converted before and store all other ones.
568  if (valueMapping.count(current))
569  continue;
570  orderedList.push_back(current);
571  // Add the current constant's dependencies to the work list. Only add
572  // constant dependencies and skip any other values such as basic block
573  // addresses.
574  for (llvm::Value *operand : current->operands())
575  if (auto *constDependency = dyn_cast<llvm::Constant>(operand))
576  workList.push_back(constDependency);
577  // Use the `getElementValue` method to add the dependencies of zero
578  // initialized aggregate constants since they do not take any operands.
579  if (auto *constAgg = dyn_cast<llvm::ConstantAggregateZero>(current)) {
580  unsigned numElements = constAgg->getElementCount().getFixedValue();
581  for (unsigned i = 0, e = numElements; i != e; ++i)
582  workList.push_back(constAgg->getElementValue(i));
583  }
584  }
585 
586  // Add the constants in reverse post order to the result set to ensure all
587  // dependencies are satisfied. Avoid storing duplicates since LLVM constants
588  // are uniqued and only one `valueMapping` entry per constant is possible.
589  SetVector<llvm::Constant *> orderedSet;
590  for (llvm::Constant *orderedConst : llvm::reverse(orderedList))

595 FailureOr<Value> ModuleImport::convertConstant(llvm::Constant *constant) {
596  // Constants have no location attached.
597  Location loc = UnknownLoc::get(context);
598 
599  // Convert constants that can be represented as attributes.
600  if (Attribute attr = getConstantAsAttr(constant)) {
601  Type type = convertType(constant->getType());
602  if (auto symbolRef = attr.dyn_cast<FlatSymbolRefAttr>()) {
603  return builder.create<AddressOfOp>(loc, type, symbolRef.getValue())
604  .getResult();
605  }
606  return builder.create<ConstantOp>(loc, type, attr).getResult();
607  }
608 
609  // Convert null pointer constants.
610  if (auto *nullPtr = dyn_cast<llvm::ConstantPointerNull>(constant)) {
611  Type type = convertType(nullPtr->getType());
612  return builder.create<NullOp>(loc, type).getResult();
613  }
614 
615  // Convert undef.
616  if (auto *undefVal = dyn_cast<llvm::UndefValue>(constant)) {
617  Type type = convertType(undefVal->getType());
618  return builder.create<UndefOp>(loc, type).getResult();
619  }
620 
621  // Convert global variable accesses.
622  if (auto *globalVar = dyn_cast<llvm::GlobalVariable>(constant)) {
623  return builder.create<AddressOfOp>(loc, processGlobal(globalVar))
624  .getResult();
625  }
626 
627  // Convert constant expressions.
628  if (auto *constExpr = dyn_cast<llvm::ConstantExpr>(constant)) {
629  // Convert the constant expression to a temporary LLVM instruction and
630  // translate it using the `processInstruction` method. Delete the
631  // instruction after the translation and remove it from `valueMapping`,
632  // since later calls to `getAsInstruction` may return the same address
633  // resulting in a conflicting `valueMapping` entry.
634  llvm::Instruction *inst = constExpr->getAsInstruction();
635  auto guard = llvm::make_scope_exit([&]() {
636  valueMapping.erase(inst);
637  inst->deleteValue();
638  });
639  // Note: `processInstruction` does not call `convertConstant` recursively
640  // since all constant dependencies have been converted before.
641  assert(llvm::all_of(inst->operands(), [&](llvm::Value *value) {
642  return valueMapping.count(value);
643  }));
644  if (failed(processInstruction(inst)))
645  return failure();
646  return lookupValue(inst);
647  }
648 
649  // Convert aggregate constants.
650  if (isa<llvm::ConstantAggregate>(constant) ||
651  isa<llvm::ConstantAggregateZero>(constant)) {
652  // Lookup the aggregate elements that have been converted before.
653  SmallVector<Value> elementValues;
654  if (auto *constAgg = dyn_cast<llvm::ConstantAggregate>(constant)) {
655  elementValues.reserve(constAgg->getNumOperands());
656  for (llvm::Value *operand : constAgg->operands())
657  elementValues.push_back(lookupValue(operand));
658  }
659  if (auto *constAgg = dyn_cast<llvm::ConstantAggregateZero>(constant)) {
660  unsigned numElements = constAgg->getElementCount().getFixedValue();
661  elementValues.reserve(numElements);
662  for (unsigned i = 0, e = numElements; i != e; ++i)
663  elementValues.push_back(lookupValue(constAgg->getElementValue(i)));
664  }
665  assert(llvm::count(elementValues, nullptr) == 0 &&
666  "expected all elements have been converted before");
667 
668  // Generate an UndefOp as root value and insert the aggregate elements.
669  Type rootType = convertType(constant->getType());
670  bool isArrayOrStruct = rootType.isa<LLVMArrayType, LLVMStructType>();
671  assert((isArrayOrStruct || LLVM::isCompatibleVectorType(rootType)) &&
672  "unrecognized aggregate type");
673  Value root = builder.create<UndefOp>(loc, rootType);
674  for (const auto &it : llvm::enumerate(elementValues)) {
675  if (isArrayOrStruct) {
676  root = builder.create<InsertValueOp>(loc, root, it.value(), it.index());
677  } else {
678  Attribute indexAttr = builder.getI32IntegerAttr(it.index());
679  Value indexValue =
680  builder.create<ConstantOp>(loc, builder.getI32Type(), indexAttr);
681  root = builder.create<InsertElementOp>(loc, rootType, root, it.value(),
682  indexValue);
683  }
684  }
685  return root;
686  }
687 
688  return emitError(loc) << "unhandled constant " << diag(*constant);
689 }

825 LogicalResult ModuleImport::convertOperation(OpBuilder &odsBuilder,
826  llvm::Instruction *inst) {
827  // Copy the operands to an LLVM operands array reference for conversion.
828  SmallVector<llvm::Value *> operands(inst->operands());
829  ArrayRef<llvm::Value *> llvmOperands(operands);
830 
831  // Convert all instructions that provide an MLIR builder.
832 #include "mlir/Dialect/LLVMIR/LLVMOpFromLLVMIRConversions.inc"
833 
834  // Convert all remaining instructions that do not provide an MLIR builder.
835  Location loc = translateLoc(inst->getDebugLoc());
836  if (inst->getOpcode() == llvm::Instruction::Br) {
837  auto *brInst = cast<llvm::BranchInst>(inst);
838 
839  SmallVector<Block *> succBlocks;
840  SmallVector<SmallVector<Value>> succBlockArgs;
841  for (auto i : llvm::seq<unsigned>(0, brInst->getNumSuccessors())) {
842  llvm::BasicBlock *succ = brInst->getSuccessor(i);
843  SmallVector<Value> blockArgs;
844  if (failed(convertBranchArgs(brInst, succ, blockArgs)))
845  return failure();
846  succBlocks.push_back(lookupBlock(succ));
847  succBlockArgs.push_back(blockArgs);
848  }
849 
850  if (brInst->isConditional()) {
851  FailureOr<Value> condition = convertValue(brInst->getCondition());
852  if (failed(condition))
853  return failure();
854  builder.create<LLVM::CondBrOp>(loc, *condition, succBlocks.front(),
855  succBlockArgs.front(), succBlocks.back(),
856  succBlockArgs.back());
857  } else {
858  builder.create<LLVM::BrOp>(loc, succBlockArgs.front(),
859  succBlocks.front());
860  }
861  return success();
862  }
863  if (inst->getOpcode() == llvm::Instruction::Switch) {
864  auto *swInst = cast<llvm::SwitchInst>(inst);
865  // Process the condition value.
866  FailureOr<Value> condition = convertValue(swInst->getCondition());
867  if (failed(condition))
868  return failure();
869  SmallVector<Value> defaultBlockArgs;
870  // Process the default case.
871  llvm::BasicBlock *defaultBB = swInst->getDefaultDest();
872  if (failed(convertBranchArgs(swInst, defaultBB, defaultBlockArgs)))
873  return failure();
874 
875  // Process the cases.
876  unsigned numCases = swInst->getNumCases();
877  SmallVector<SmallVector<Value>> caseOperands(numCases);
878  SmallVector<ValueRange> caseOperandRefs(numCases);
879  SmallVector<int32_t> caseValues(numCases);
880  SmallVector<Block *> caseBlocks(numCases);
881  for (const auto &it : llvm::enumerate(swInst->cases())) {
882  const llvm::SwitchInst::CaseHandle &caseHandle = it.value();
883  llvm::BasicBlock *succBB = caseHandle.getCaseSuccessor();
884  if (failed(convertBranchArgs(swInst, succBB, caseOperands[it.index()])))
885  return failure();
886  caseOperandRefs[it.index()] = caseOperands[it.index()];
887  caseValues[it.index()] = caseHandle.getCaseValue()->getSExtValue();
888  caseBlocks[it.index()] = lookupBlock(succBB);
889  }
890 
891  builder.create<SwitchOp>(loc, *condition, lookupBlock(defaultBB),
892  defaultBlockArgs, caseValues, caseBlocks,
893  caseOperandRefs);
894  return success();
895  }
896  if (inst->getOpcode() == llvm::Instruction::PHI) {
897  Type type = convertType(inst->getType());
898  mapValue(inst, builder.getInsertionBlock()->addArgument(
899  type, translateLoc(inst->getDebugLoc())));
900  return success();
901  }
902  if (inst->getOpcode() == llvm::Instruction::Call) {
903  auto *callInst = cast<llvm::CallInst>(inst);
904 
905  SmallVector<Type> types;
906  SmallVector<Value> operands;
907  if (failed(convertCallTypeAndOperands(callInst, types, operands)))
908  return failure();
909 
910  CallOp callOp;
911  if (llvm::Function *callee = callInst->getCalledFunction()) {
912  callOp = builder.create<CallOp>(
913  loc, types, SymbolRefAttr::get(context, callee->getName()), operands);
914  } else {
915  callOp = builder.create<CallOp>(loc, types, operands);
916  }
917  setFastmathFlagsAttr(inst, callOp);
918  if (!callInst->getType()->isVoidTy())
919  mapValue(inst, callOp.getResult());
920  return success();
921  }
922  if (inst->getOpcode() == llvm::Instruction::LandingPad) {
923  auto *lpInst = cast<llvm::LandingPadInst>(inst);
924 
925  SmallVector<Value> operands;
926  operands.reserve(lpInst->getNumClauses());
927  for (auto i : llvm::seq<unsigned>(0, lpInst->getNumClauses())) {
928  FailureOr<Value> operand = convertConstantExpr(lpInst->getClause(i));
929  if (failed(operand))
930  return failure();
931  operands.push_back(*operand);
932  }
933 
934  Type type = convertType(lpInst->getType());
935  Value res =
936  builder.create<LandingpadOp>(loc, type, lpInst->isCleanup(), operands);
937  mapValue(inst, res);
938  return success();
939  }
940  if (inst->getOpcode() == llvm::Instruction::Invoke) {
941  auto *invokeInst = cast<llvm::InvokeInst>(inst);
942 
943  SmallVector<Type> types;
944  SmallVector<Value> operands;
945  if (failed(convertCallTypeAndOperands(invokeInst, types, operands)))
946  return failure();
947 
948  SmallVector<Value> normalArgs, unwindArgs;
949  (void)convertBranchArgs(invokeInst, invokeInst->getNormalDest(),
950  normalArgs);
951  (void)convertBranchArgs(invokeInst, invokeInst->getUnwindDest(),
952  unwindArgs);
953 
954  InvokeOp invokeOp;
955  if (llvm::Function *callee = invokeInst->getCalledFunction()) {
956  invokeOp = builder.create<InvokeOp>(
957  loc, types,
958  SymbolRefAttr::get(builder.getContext(), callee->getName()), operands,
959  lookupBlock(invokeInst->getNormalDest()), normalArgs,
960  lookupBlock(invokeInst->getUnwindDest()), unwindArgs);
961  } else {
962  invokeOp = builder.create<InvokeOp>(
963  loc, types, operands, lookupBlock(invokeInst->getNormalDest()),
964  normalArgs, lookupBlock(invokeInst->getUnwindDest()), unwindArgs);
965  }
966  if (!invokeInst->getType()->isVoidTy())
967  mapValue(inst, invokeOp.getResults().front());
968  return success();
969  }
970  if (inst->getOpcode() == llvm::Instruction::GetElementPtr) {
971  auto *gepInst = cast<llvm::GetElementPtrInst>(inst);
972  Type sourceElementType = convertType(gepInst->getSourceElementType());
973  FailureOr<Value> basePtr = convertValue(gepInst->getOperand(0));
974  if (failed(basePtr))
975  return failure();
976 
977  // Treat every indices as dynamic since GEPOp::build will refine those
978  // indices into static attributes later. One small downside of this
979  // approach is that many unused `llvm.mlir.constant` would be emitted
980  // at first place.
981  SmallVector<GEPArg> indices;
982  for (llvm::Value *operand : llvm::drop_begin(gepInst->operand_values())) {
983  FailureOr<Value> index = convertValue(operand);
984  if (failed(index))
985  return failure();
986  indices.push_back(*index);
987  }
988 
989  Type type = convertType(inst->getType());
990  Value res = builder.create<GEPOp>(loc, type, sourceElementType, *basePtr,
991  indices, gepInst->isInBounds());
992  mapValue(inst, res);
993  return success();
994  }
995 
996  return emitError(loc) << "unhandled instruction " << diag(*inst);
997 }
