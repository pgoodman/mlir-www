20 struct ExpandShapeOpInterface
21  : public RuntimeVerifiableOpInterface::ExternalModel<ExpandShapeOpInterface,
22  ExpandShapeOp> {
23  void generateRuntimeVerification(Operation *op, OpBuilder &builder,
24  Location loc) const {
25  auto expandShapeOp = cast<ExpandShapeOp>(op);
26 
27  // Verify that the expanded dim sizes are a product of the collapsed dim
28  // size.
29  for (auto it : llvm::enumerate(expandShapeOp.getReassociationIndices())) {
30  Value srcDimSz =
31  builder.create<DimOp>(loc, expandShapeOp.getSrc(), it.index());
32  int64_t groupSz = 1;
33  bool foundDynamicDim = false;
34  for (int64_t resultDim : it.value()) {
35  if (expandShapeOp.getResultType().isDynamicDim(resultDim)) {
36  // Keep this assert here in case the op is extended in the future.
37  assert(!foundDynamicDim &&
38  "more than one dynamic dim found in reassoc group");
39  (void)foundDynamicDim;
40  foundDynamicDim = true;
41  continue;
42  }
43  groupSz *= expandShapeOp.getResultType().getDimSize(resultDim);
44  }
45  Value staticResultDimSz =
46  builder.create<arith::ConstantIndexOp>(loc, groupSz);
47  // staticResultDimSz must divide srcDimSz evenly.
48  Value mod =
49  builder.create<arith::RemSIOp>(loc, srcDimSz, staticResultDimSz);
50  Value isModZero = builder.create<arith::CmpIOp>(
51  loc, arith::CmpIPredicate::eq, mod,
52  builder.create<arith::ConstantIndexOp>(loc, 0));
53  builder.create<cf::AssertOp>(
54  loc, isModZero,
55  "static result dims in reassoc group do not divide src dim evenly");
56  }
57  }
58 };
59 } // namespace
60 } // namespace memref
61 } // namespace mlir
