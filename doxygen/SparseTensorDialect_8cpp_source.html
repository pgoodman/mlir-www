74 Attribute SparseTensorEncodingAttr::parse(AsmParser &parser, Type type) {
75  if (failed(parser.parseLess()))
76  return {};
77  // Parse the data as a dictionary.
78  DictionaryAttr dict;
79  if (failed(parser.parseAttribute(dict)))
80  return {};
81  if (failed(parser.parseGreater()))
82  return {};
83  // Process the data from the parsed dictionary value into struct-like data.
84  SmallVector<DimLevelType> dlt;
85  AffineMap dimOrd = {};
86  AffineMap higherOrd = {};
87  unsigned ptr = 0;
88  unsigned ind = 0;
89  for (const NamedAttribute &attr : dict) {
90  if (attr.getName() == "dimLevelType") {
91  auto arrayAttr = attr.getValue().dyn_cast<ArrayAttr>();
92  if (!arrayAttr) {
93  parser.emitError(parser.getNameLoc(),
94  "expected an array for dimension level types");
95  return {};
96  }
97  for (auto i : arrayAttr) {
98  auto strAttr = i.dyn_cast<StringAttr>();
99  if (!strAttr) {
100  parser.emitError(parser.getNameLoc(),
101  "expected a string value in dimension level types");
102  return {};
103  }
104  auto strVal = strAttr.getValue();
105  if (strVal == "dense") {
106  dlt.push_back(DimLevelType::Dense);
107  } else if (strVal == "compressed") {
108  dlt.push_back(DimLevelType::Compressed);
109  } else if (strVal == "compressed-nu") {
110  dlt.push_back(DimLevelType::CompressedNu);
111  } else if (strVal == "compressed-no") {
112  dlt.push_back(DimLevelType::CompressedNo);
113  } else if (strVal == "compressed-nu-no") {
114  dlt.push_back(DimLevelType::CompressedNuNo);
115  } else if (strVal == "singleton") {
116  dlt.push_back(DimLevelType::Singleton);
117  } else if (strVal == "singleton-nu") {
118  dlt.push_back(DimLevelType::SingletonNu);
119  } else if (strVal == "singleton-no") {
120  dlt.push_back(DimLevelType::SingletonNo);
121  } else if (strVal == "singleton-nu-no") {
122  dlt.push_back(DimLevelType::SingletonNuNo);
123  } else {
124  parser.emitError(parser.getNameLoc(),
125  "unexpected dimension level type: ")
126  << strVal;
127  return {};
128  }
129  }
130  } else if (attr.getName() == "dimOrdering") {
131  auto affineAttr = attr.getValue().dyn_cast<AffineMapAttr>();
132  if (!affineAttr) {
133  parser.emitError(parser.getNameLoc(),
134  "expected an affine map for dimension ordering");
135  return {};
136  }
137  dimOrd = affineAttr.getValue();
138  } else if (attr.getName() == "higherOrdering") {
139  auto affineAttr = attr.getValue().dyn_cast<AffineMapAttr>();
140  if (!affineAttr) {
141  parser.emitError(parser.getNameLoc(),
142  "expected an affine map for higher ordering");
143  return {};
144  }
145  higherOrd = affineAttr.getValue();
146  } else if (attr.getName() == "pointerBitWidth") {
147  auto intAttr = attr.getValue().dyn_cast<IntegerAttr>();
148  if (!intAttr) {
149  parser.emitError(parser.getNameLoc(),
150  "expected an integral pointer bitwidth");
151  return {};
152  }
153  ptr = intAttr.getInt();
154  } else if (attr.getName() == "indexBitWidth") {
155  auto intAttr = attr.getValue().dyn_cast<IntegerAttr>();
156  if (!intAttr) {
157  parser.emitError(parser.getNameLoc(),
158  "expected an integral index bitwidth");
159  return {};
160  }
161  ind = intAttr.getInt();
162  } else {
163  parser.emitError(parser.getNameLoc(), "unexpected key: ")
164  << attr.getName().strref();
165  return {};
166  }
167  }
168  // Construct struct-like storage for attribute.
169  return parser.getChecked<SparseTensorEncodingAttr>(
170  parser.getContext(), dlt, dimOrd, higherOrd, ptr, ind);
171 }

194 LogicalResult SparseTensorEncodingAttr::verify(
195  function_ref<InFlightDiagnostic()> emitError,
196  ArrayRef<DimLevelType> dimLevelType, AffineMap dimOrdering,
197  AffineMap higherOrdering, unsigned pointerBitWidth,
198  unsigned indexBitWidth) {
199  if (!acceptBitWidth(pointerBitWidth))
200  return emitError() << "unexpected pointer bitwidth: " << pointerBitWidth;
201  if (!acceptBitWidth(indexBitWidth))
202  return emitError() << "unexpected index bitwidth: " << indexBitWidth;
203  if (dimOrdering) {
204  if (!dimOrdering.isPermutation())
205  return emitError()
206  << "expected a permutation affine map for dimension ordering";
207  if (dimOrdering.getNumResults() != dimLevelType.size())
208  return emitError() << "unexpected mismatch in ordering and dimension "
209  "level types size";
210  }
211  if (higherOrdering) {
212  if (higherOrdering.getNumDims() >= higherOrdering.getNumResults())
213  return emitError() << "unexpected higher ordering mapping from "
214  << higherOrdering.getNumDims() << " to "
215  << higherOrdering.getNumResults();
216  if (higherOrdering.getNumResults() != dimLevelType.size())
217  return emitError() << "unexpected mismatch in higher ordering and "
218  "dimension level types size";
219  }
220  return success();
221 }

223 LogicalResult SparseTensorEncodingAttr::verifyEncoding(
224  ArrayRef<int64_t> shape, Type elementType,
225  function_ref<InFlightDiagnostic()> emitError) const {
226  // Check structural integrity.
227  if (failed(verify(emitError, getDimLevelType(), getDimOrdering(),
228  getHigherOrdering(), getPointerBitWidth(),
229  getIndexBitWidth())))
230  return failure();
231  // Check integrity with tensor type specifics. Dimension ordering is optional,
232  // but we always should have dimension level types for the full rank.
233  unsigned size = shape.size();
234  if (size == 0)
235  return emitError() << "expected non-scalar sparse tensor";
236  if (getHigherOrdering()) {
237  if (getHigherOrdering().getNumDims() != size)
238  return emitError() << "expected an affine map of size " << size
239  << " for higher ordering";
240 
241  // TODO: verification of higher ordering contents
242 
243  size = getHigherOrdering().getNumResults(); // higher-order size!
244  }
245  if (getDimOrdering() && getDimOrdering().getNumResults() != size)
246  return emitError() << "expected an affine map of size " << size
247  << " for dimension ordering";
248  if (getDimLevelType().size() != size)
249  return emitError() << "expected an array of size " << size
250  << " for dimension level types";
251  return success();
252 }

326 /// We normalized sparse tensor encoding attribute by always using
327 /// ordered/unique DLT such that "compressed-nu-no" and "compressed-nu" (as well
328 /// as other variants) lead to the same storage specifier type, and stripping
329 /// irrelevant fields that does not alter the sparse tensor memory layout.
330 static SparseTensorEncodingAttr
331 getNormalizedEncodingForSpecifier(SparseTensorEncodingAttr enc) {
332  SmallVector<DimLevelType> dlts;
333  for (auto dlt : enc.getDimLevelType())
334  dlts.push_back(*getDimLevelType(*getLevelFormat(dlt), true, true));
335 
336  return SparseTensorEncodingAttr::get(
337  enc.getContext(), dlts,
338  AffineMap(), // dimOrdering (irrelavant to storage speicifer)
339  AffineMap(), // highLvlOrdering (irrelavant to storage specifer)
340  enc.getPointerBitWidth(), enc.getIndexBitWidth());
341 }

394 static LogicalResult verifySparsifierGetterSetter(
395  StorageSpecifierKind mdKind, std::optional<APInt> dim,
396  TypedValue<StorageSpecifierType> md, Operation *op) {
397  if (mdKind == StorageSpecifierKind::ValMemSize && dim) {
398  return op->emitError(
399  "redundant dimension argument for querying value memory size");
400  }
401 
402  auto enc = md.getType().getEncoding();
403  ArrayRef<DimLevelType> dlts = enc.getDimLevelType();
404  unsigned rank = dlts.size();
405 
406  if (mdKind != StorageSpecifierKind::ValMemSize) {
407  if (!dim)
408  return op->emitError("missing dimension argument");
409 
410  unsigned d = dim.value().getZExtValue();
411  if (d >= rank)
412  return op->emitError("requested dimension out of bound");
413 
414  if (mdKind == StorageSpecifierKind::PtrMemSize && isSingletonDLT(dlts[d]))
415  return op->emitError(
416  "requested pointer memory size on a singleton level");
417  }
418  return success();
419 }

428 LogicalResult ConvertOp::verify() {
429  if (auto tp1 = getSource().getType().dyn_cast<RankedTensorType>()) {
430  if (auto tp2 = getDest().getType().dyn_cast<RankedTensorType>()) {
431  if (tp1.getRank() != tp2.getRank())
432  return emitError("unexpected conversion mismatch in rank");
433  auto shape1 = tp1.getShape();
434  auto shape2 = tp2.getShape();
435  // Accept size matches between the source and the destination type
436  // (e.g. 10 vs. 10, 10 vs. ?, or ? vs. ?), but reject direct mismatches or
437  // matches that would need a runtime assert (e.g. 10 vs. 20 or ? vs. 10).
438  for (unsigned d = 0, rank = tp1.getRank(); d < rank; d++)
439  if (shape1[d] != shape2[d] && shape2[d] != ShapedType::kDynamic)
440  return emitError("unexpected conversion mismatch in dimension ") << d;
441  return success();
442  }
443  }
444  return emitError("unexpected type in convert");
445 }

532 template <class T>
533 static LogicalResult verifyNumBlockArgs(T *op, Region &region,
534  const char *regionName,
535  TypeRange inputTypes, Type outputType) {
536  unsigned numArgs = region.getNumArguments();
537  unsigned expectedNum = inputTypes.size();
538  if (numArgs != expectedNum)
539  return op->emitError() << regionName << " region must have exactly "
540  << expectedNum << " arguments";
541 
542  for (unsigned i = 0; i < numArgs; i++) {
543  Type typ = region.getArgument(i).getType();
544  if (typ != inputTypes[i])
545  return op->emitError() << regionName << " region argument " << (i + 1)
546  << " type mismatch";
547  }
548  Operation *term = region.front().getTerminator();
549  YieldOp yield = dyn_cast<YieldOp>(term);
550  if (!yield)
551  return op->emitError() << regionName
552  << " region must end with sparse_tensor.yield";
553  if (!yield.getResult() || yield.getResult().getType() != outputType)
554  return op->emitError() << regionName << " region yield type mismatch";
555 
556  return success();
557 }

559 LogicalResult BinaryOp::verify() {
560  NamedAttrList attrs = (*this)->getAttrs();
561  Type leftType = getX().getType();
562  Type rightType = getY().getType();
563  Type outputType = getOutput().getType();
564  Region &overlap = getOverlapRegion();
565  Region &left = getLeftRegion();
566  Region &right = getRightRegion();
567 
568  // Check correct number of block arguments and return type for each
569  // non-empty region.
570  LogicalResult regionResult = success();
571  if (!overlap.empty()) {
572  regionResult = verifyNumBlockArgs(
573  this, overlap, "overlap", TypeRange{leftType, rightType}, outputType);
574  if (failed(regionResult))
575  return regionResult;
576  }
577  if (!left.empty()) {
578  regionResult =
579  verifyNumBlockArgs(this, left, "left", TypeRange{leftType}, outputType);
580  if (failed(regionResult))
581  return regionResult;
582  } else if (getLeftIdentity()) {
583  if (leftType != outputType)
584  return emitError("left=identity requires first argument to have the same "
585  "type as the output");
586  }
587  if (!right.empty()) {
588  regionResult = verifyNumBlockArgs(this, right, "right",
589  TypeRange{rightType}, outputType);
590  if (failed(regionResult))
591  return regionResult;
592  } else if (getRightIdentity()) {
593  if (rightType != outputType)
594  return emitError("right=identity requires second argument to have the "
595  "same type as the output");
596  }
597 
598  return success();
599 }

642  if (concatDim >= rank)
643  return emitError(llvm::formatv(
644  "Failed to concatentate tensors with rank={0} on dimension={1}.", rank,
645  concatDim));
646 
647  for (size_t i = 0, e = getInputs().size(); i < e; i++) {
648  Value input = getInputs()[i];
649  auto inputRank = input.getType().cast<RankedTensorType>().getRank();
650  if (inputRank != rank)
651  return emitError(
652  llvm::formatv("The input tensor ${0} has a different rank (rank={1}) "
653  "from the output tensor (rank={2}).",
654  i, inputRank, rank));
655  }
656 
657  for (unsigned i = 0; i < rank; i++) {
658  auto dstDim = dstTp.getShape()[i];
659  if (i == concatDim) {
660  if (!ShapedType::isDynamic(dstDim)) {
661  unsigned sumDim = 0;
662  for (auto src : getInputs()) {
663  // If we reach here, all inputs should have static shapes.
664  auto d = src.getType().cast<RankedTensorType>().getShape()[i];
665  sumDim += d;
666  }
667  // If all dimension are statically known, the sum of all the input
668  // dimensions should be equal to the output dimension.
669  if (sumDim != dstDim)
670  return emitError(
671  "The concatenation dimension of the output tensor should be the "
672  "sum of all the concatenation dimensions of the input tensors.");
673  }
674  } else {
675  int64_t prev = dstDim;
676  for (auto src : getInputs()) {
677  auto d = src.getType().cast<RankedTensorType>().getShape()[i];
678  if (!ShapedType::isDynamic(prev) && d != prev)
679  return emitError("All dimensions (expect for the concatenating one) "
680  "should be equal.");
681  prev = d;
682  }
683  }
684  }
685 
686  return success();
687 }

725 void ForeachOp::build(
726  OpBuilder &builder, OperationState &result, Value tensor,
727  ValueRange initArgs,
728  function_ref<void(OpBuilder &, Location, ValueRange, Value, ValueRange)>
729  bodyBuilder) {
730  build(builder, result, initArgs.getTypes(), tensor, initArgs);
731  // Builds foreach body.
732  if (!bodyBuilder)
733  return;
734  auto rtp = tensor.getType().cast<RankedTensorType>();
735  int64_t rank = rtp.getRank();
736 
737  SmallVector<Type> blockArgTypes;
738  // Starts with n index.
739  std::fill_n(std::back_inserter(blockArgTypes), rank, builder.getIndexType());
740  // Followed by one value.
741  blockArgTypes.push_back(rtp.getElementType());
742  // Followed by reduction variable.
743  blockArgTypes.append(initArgs.getTypes().begin(), initArgs.getTypes().end());
744 
745  SmallVector<Location> blockArgLocs;
746  std::fill_n(std::back_inserter(blockArgLocs), blockArgTypes.size(),
747  tensor.getLoc());
748 
749  OpBuilder::InsertionGuard guard(builder);
750  auto &region = *result.regions.front();
751  Block *bodyBlock =
752  builder.createBlock(&region, region.end(), blockArgTypes, blockArgLocs);
753  bodyBuilder(builder, result.location,
754  bodyBlock->getArguments().slice(0, rank),
755  bodyBlock->getArguments()[rank],
756  bodyBlock->getArguments().drop_front(rank + 1));
757 }

825  return emitError("need at least one xs buffer.");

827  auto n = getN().getDefiningOp<arith::ConstantIndexOp>();
828

831  bool checkEleType = true) -> LogicalResult {

838  return emitError(llvm::formatv("xs and ys need to have a dimension >= n"
839  ": {0} < {1}",

841 
842  if (checkEleType && xtp != mtp.getElementType())
843  return emitError("mismatch xs element types");
844  }
845  return success();
846  };
847 
848  LogicalResult result = checkTypes(getXs());
849  if (failed(result))
850  return result;
851 
852  if (n)
853  return checkTypes(getYs(), false);
854 
855  return success();
