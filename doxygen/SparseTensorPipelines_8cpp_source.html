27 /// Return configuration options for One-Shot Bufferize.
28 static bufferization::OneShotBufferizationOptions
29 getBufferizationOptions(bool analysisOnly) {
30  using namespace bufferization;
31  OneShotBufferizationOptions options;
32  options.bufferizeFunctionBoundaries = true;
33  // TODO(springerm): To spot memory leaks more easily, returning dense allocs
34  // should be disallowed.
35  options.allowReturnAllocs = true;
36  options.functionBoundaryTypeConversion = LayoutMapOption::IdentityLayoutMap;
37  options.unknownTypeConverterFn = [](Value value, Attribute memorySpace,
38  const BufferizationOptions &options) {
39  return getMemRefTypeWithStaticIdentityLayout(
40  value.getType().cast<TensorType>(), memorySpace);
41  };
42  if (analysisOnly) {
43  options.testAnalysisOnly = true;
44  options.printConflicts = true;
45  }
46  return options;
47 }

53 void mlir::sparse_tensor::buildSparseCompiler(
54  OpPassManager &pm, const SparseCompilerOptions &options) {
55  pm.addNestedPass<func::FuncOp>(createLinalgGeneralizationPass());
56  pm.addPass(createSparsificationAndBufferizationPass(
57  getBufferizationOptions(options.testBufferizationAnalysisOnly),
58  options.sparsificationOptions(), options.sparseTensorConversionOptions(),
59  options.enableRuntimeLibrary, options.enableBufferInitialization,
60  options.vectorLength,
61  /*enableVLAVectorization=*/options.armSVE,
62  /*enableSIMDIndex32=*/options.force32BitVectorIndices));
63  if (options.testBufferizationAnalysisOnly)
64  return;
65  pm.addNestedPass<func::FuncOp>(createCanonicalizerPass());
66  pm.addNestedPass<func::FuncOp>(
67  mlir::bufferization::createFinalizingBufferizePass());
68  // TODO(springerm): Add sparse support to the BufferDeallocation pass and add
69  // it to this pipeline.
70  pm.addNestedPass<func::FuncOp>(createConvertLinalgToLoopsPass());
71  pm.addNestedPass<func::FuncOp>(createConvertVectorToSCFPass());
72  pm.addNestedPass<func::FuncOp>(createConvertSCFToCFPass());
73  pm.addPass(memref::createExpandStridedMetadataPass());
74  pm.addPass(createLowerAffinePass());
75  pm.addPass(createConvertVectorToLLVMPass(options.lowerVectorToLLVMOptions()));
76  pm.addPass(createMemRefToLLVMConversionPass());
77  pm.addNestedPass<func::FuncOp>(createConvertComplexToStandardPass());
78  pm.addNestedPass<mlir::func::FuncOp>(mlir::arith::createArithExpandOpsPass());
79  pm.addNestedPass<func::FuncOp>(createConvertMathToLLVMPass());
80  pm.addPass(createConvertMathToLibmPass());
81  pm.addPass(createConvertComplexToLibmPass());
82  // Repeat convert-vector-to-llvm.
83  pm.addPass(createConvertVectorToLLVMPass(options.lowerVectorToLLVMOptions()));
84  pm.addPass(createConvertComplexToLLVMPass());
85  pm.addPass(createConvertVectorToLLVMPass(options.lowerVectorToLLVMOptions()));
86  pm.addPass(createConvertFuncToLLVMPass());
87  pm.addPass(createReconcileUnrealizedCastsPass());
88 }
