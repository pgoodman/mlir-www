33 convertSparseTensorType(RankedTensorType rtp, SmallVectorImpl<Type> &fields) {
34  auto enc = getSparseTensorEncoding(rtp);
35  if (!enc)
36  return std::nullopt;

38  foreachFieldAndTypeInSparseTensor(
39  rtp,

41  SparseTensorFieldKind /*fieldKind*/, unsigned /*dim*/,

46  });
47  return success();

117 constexpr uint64_t kDataFieldStartingIdx = 0;
118 
119 void sparse_tensor::foreachFieldInSparseTensor(
120  const SparseTensorEncodingAttr enc,
121  llvm::function_ref<bool(unsigned, SparseTensorFieldKind, unsigned,
122  DimLevelType)>
123  callback) {
124  assert(enc);
125 
126 #define RETURN_ON_FALSE(idx, kind, dim, dlt) \
127  if (!(callback(idx, kind, dim, dlt))) \
128  return;
129 
130  static_assert(kDataFieldStartingIdx == 0);
131  unsigned fieldIdx = kDataFieldStartingIdx;
132  // Per-dimension storage.
133  for (unsigned r = 0, rank = enc.getDimLevelType().size(); r < rank; r++) {
134  // Dimension level types apply in order to the reordered dimension.
135  // As a result, the compound type can be constructed directly in the given
136  // order.
137  auto dlt = getDimLevelType(enc, r);
138  if (isCompressedDLT(dlt)) {
139  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::PtrMemRef, r, dlt);
140  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::IdxMemRef, r, dlt);
141  } else if (isSingletonDLT(dlt)) {
142  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::IdxMemRef, r, dlt);
143  } else {
144  assert(isDenseDLT(dlt)); // no fields
145  }
146  }
147  // The values array.
148  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::ValMemRef, -1u,
149  DimLevelType::Undef);
150 
151  // Put metadata at the end.
152  RETURN_ON_FALSE(fieldIdx++, SparseTensorFieldKind::StorageSpec, -1u,
153  DimLevelType::Undef);
154 
155 #undef RETURN_ON_FALSE
156 }

158 void sparse_tensor::foreachFieldAndTypeInSparseTensor(
159  RankedTensorType rType,
160  llvm::function_ref<bool(Type, unsigned, SparseTensorFieldKind, unsigned,
161  DimLevelType)>
162  callback) {
163  auto enc = getSparseTensorEncoding(rType);
164  assert(enc);
165  // Construct the basic types.
166  Type idxType = enc.getIndexType();
167  Type ptrType = enc.getPointerType();
168  Type eltType = rType.getElementType();
169 
170  Type metaDataType = StorageSpecifierType::get(enc);
171  // memref<? x ptr> pointers
172  Type ptrMemType = MemRefType::get({ShapedType::kDynamic}, ptrType);
173  // memref<? x idx> indices
174  Type idxMemType = MemRefType::get({ShapedType::kDynamic}, idxType);
175  // memref<? x eltType> values
176  Type valMemType = MemRefType::get({ShapedType::kDynamic}, eltType);
177 
178  foreachFieldInSparseTensor(
179  enc,
180  [metaDataType, ptrMemType, idxMemType, valMemType,
181  callback](unsigned fieldIdx, SparseTensorFieldKind fieldKind,
182  unsigned dim, DimLevelType dlt) -> bool {
183  switch (fieldKind) {
184  case SparseTensorFieldKind::StorageSpec:
185  return callback(metaDataType, fieldIdx, fieldKind, dim, dlt);
186  case SparseTensorFieldKind::PtrMemRef:
187  return callback(ptrMemType, fieldIdx, fieldKind, dim, dlt);
188  case SparseTensorFieldKind::IdxMemRef:
189  return callback(idxMemType, fieldIdx, fieldKind, dim, dlt);
190  case SparseTensorFieldKind::ValMemRef:
191  return callback(valMemType, fieldIdx, fieldKind, dim, dlt);
192  };
193  llvm_unreachable("unrecognized field kind");
194  });
195 }

208 unsigned
209 sparse_tensor::getNumDataFieldsFromEncoding(SparseTensorEncodingAttr enc) {
210  unsigned numFields = 0; // one value memref
211  foreachFieldInSparseTensor(enc,
212  [&numFields](unsigned fidx, SparseTensorFieldKind,
213  unsigned, DimLevelType) -> bool {
214  if (fidx >= kDataFieldStartingIdx)
215  numFields++;
216  return true;
217  });
218  numFields -= 1; // the last field is MetaData field
219  assert(numFields ==

221  return numFields;
