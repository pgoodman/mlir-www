26 // SparseTensorDescriptor and helpers, manage the sparse tensor memory layout
27 // scheme.
28 //
29 // Sparse tensor storage scheme for rank-dimensional tensor is organized
30 // as a single compound type with the following fields. Note that every
31 // memref with ? size actually behaves as a "vector", i.e. the stored
32 // size is the capacity and the used size resides in the memSizes array.
33 //
34 // struct {
35 // ; per-dimension d:
36 // ; if dense:
37 // <nothing>
38 // ; if compresed:
39 // memref<? x ptr> pointers-d ; pointers for sparse dim d
40 // memref<? x idx> indices-d ; indices for sparse dim d
41 // ; if singleton:
42 // memref<? x idx> indices-d ; indices for singleton dim d
43 // memref<? x eltType> values ; values
44 //
45 // ; sparse tensor metadata
46 // struct {
47 // array<rank x int> dimSizes ; sizes for each dimension
48 // array<n x int> memSizes; ; sizes for each data memref
49 // }
50 // };
51 //
52 //===----------------------------------------------------------------------===//
53 enum class SparseTensorFieldKind : uint32_t {
54  StorageSpec = 0,
55  PtrMemRef = 1,
56  IdxMemRef = 2,
57  ValMemRef = 3
58 };

60 static_assert(static_cast<uint32_t>(SparseTensorFieldKind::PtrMemRef) ==

62 static_assert(static_cast<uint32_t>(SparseTensorFieldKind::IdxMemRef) ==

64 static_assert(static_cast<uint32_t>(SparseTensorFieldKind::ValMemRef) ==

67 /// For each field that will be allocated for the given sparse tensor encoding,
68 /// calls the callback with the corresponding field index, field kind, dimension
69 /// (for sparse tensor level memrefs) and dimlevelType.
70 /// The field index always starts with zero and increments by one between two
71 /// callback invocations.
72 /// Ideally, all other methods should rely on this function to query a sparse
73 /// tensor fields instead of relying on ad-hoc index computation.

75  SparseTensorEncodingAttr,
76  llvm::function_ref<bool(unsigned /*fieldIdx*/,
77  SparseTensorFieldKind /*fieldKind*/,
78  unsigned /*dim (if applicable)*/,

149 /// A helper class around an array of values that corresponding to a sparse
150 /// tensor, provides a set of meaningful APIs to query and update a particular
151 /// field in a consistent way.
152 /// Users should not make assumption on how a sparse tensor is laid out but
153 /// instead relies on this class to access the right value for the right field.
154 template <bool mut>
155 class SparseTensorDescriptorImpl {
156 private:
157  // Uses ValueRange for immuatable descriptors; uses SmallVectorImpl<Value> &
158  // for mutable descriptors.
159  // Using SmallVector for mutable descriptor allows users to reuse it as a tmp
160  // buffers to append value for some special cases, though users should be
161  // responsible to restore the buffer to legal states after their use. It is
162  // probably not a clean way, but it is the most efficient way to avoid copying
163  // the fields into another SmallVector. If a more clear way is wanted, we
164  // should change it to MutableArrayRef instead.
165  using ValueArrayRef = typename std::conditional<mut, SmallVectorImpl<Value> &,
166  ValueRange>::type;
167 
168 public:
169  SparseTensorDescriptorImpl(Type tp, ValueArrayRef fields)
170  : rType(tp.cast<RankedTensorType>()), fields(fields) {
171  assert(getSparseTensorEncoding(tp) &&
172  getNumFieldsFromEncoding(getSparseTensorEncoding(tp)) ==
173  fields.size());
174  // We should make sure the class is trivially copyable (and should be small
175  // enough) such that we can pass it by value.
176  static_assert(
177  std::is_trivially_copyable_v<SparseTensorDescriptorImpl<mut>>);
178  }
