90 /// Determines if affine expression is invariant.
91 static bool isInvariantAffine(AffineExpr a, ArrayRef<unsigned> loopStack,
92  unsigned ldx, bool &atLevel) {
93  switch (a.getKind()) {
94  case AffineExprKind::DimId: {
95  unsigned idx = a.cast<AffineDimExpr>().getPosition();
96  if (idx == ldx) {
97  atLevel = true;
98  // Must be invariant if we are at the level.
99  return true;
100  }
101  bool isInvariant = false;
102  for (unsigned loop : loopStack) {
103  isInvariant = (loop == idx);
104  if (isInvariant)
105  break;
106  }
107  return isInvariant;
108  }
109  case AffineExprKind::Add:
110  case AffineExprKind::Mul: {
111  auto binOp = a.cast<AffineBinaryOpExpr>();
112  return isInvariantAffine(binOp.getLHS(), loopStack, ldx, atLevel) &&
113  isInvariantAffine(binOp.getRHS(), loopStack, ldx, atLevel);
114  }
115  default: {
116  assert(a.isa<AffineConstantExpr>());
117  return true;
118  }
119  }
120 }

128 /// Helper method to construct a permuted dimension ordering
129 /// that adheres to the given topological sort.
130 static AffineMap permute(CodegenEnv &env, AffineMap m) {
131  assert(m.getNumDims() + env.merger().getNumFilterLoops() ==
132  env.topSortSize() &&
133  "size mismatch");
134  // Construct the inverse of `m`; to avoid the asymptotic complexity
135  // of calling `m.getPermutedPosition` repeatedly.
136  SmallVector<unsigned> perm;
137  unsigned numResults = m.getNumResults();
138  BitVector worklist(numResults, true);
139  unsigned loopDepth = 1;
140 
141  // Construct the permutation.
142  while (worklist.any() && loopDepth <= env.topSortSize()) {
143  unsigned preSize = perm.size();
144  for (auto dim : worklist.set_bits()) {
145  bool atLevel = false;
146  if (m.getResult(dim).isa<AffineConstantExpr>() ||
147  (isInvariantAffine(m.getResult(dim),
148  env.getTopSortSlice(0, loopDepth),
149  env.topSortAt(loopDepth - 1), atLevel) &&
150  atLevel)) {
151  // If the matching affine is constant expression or just become
152  // invariant. We can visit the dimension now without breaking the
153  // topSort constraint.
154  perm.push_back(dim);
155  }
156  }
157 
158  // Removes resolved dimension.
159  for (unsigned i = preSize, e = perm.size(); i < e; i++)
160  worklist.reset(perm[i]);
161 
162  // Tries to entering the next loop level.
163  loopDepth += 1;
164  }
165 
166  assert(perm.size() == numResults);
167  return AffineMap::getPermutationMap(perm, env.op().getContext());
168 }

170 /// Helper method to inspect affine expressions. Rejects cases where the
171 /// same index is used more than once. Also rejects compound affine
172 /// expressions in sparse dimensions.
173 /// filterIdx stores the current filter loop idx should be used for the next
174 /// compound affine sparse level, and it will be incremented by one when
175 /// used.
176 static bool findAffine(Merger &merger, unsigned tensor, unsigned dim,
177  AffineExpr a, DimLevelType dlt, unsigned &filterLdx,
178  bool setLvlFormat = true) {
179  switch (a.getKind()) {
180  case AffineExprKind::DimId: {
181  unsigned idx = a.cast<AffineDimExpr>().getPosition();
182  if (!isUndefDLT(merger.getDimLevelType(tensor, idx)))
183  return false; // used more than once
184 
185  if (setLvlFormat)
186  merger.setDimAndDimLevelType(tensor, idx, dim, dlt);
187  return true;
188  }
189  case AffineExprKind::Add:
190  case AffineExprKind::Mul:
191  case AffineExprKind::Constant: {
192  if (!isDenseDLT(dlt) && setLvlFormat) {
193  assert(isUndefDLT(merger.getDimLevelType(tensor, filterLdx)));
194  // Use a filter loop for sparse affine expression.
195  merger.setDimAndDimLevelType(tensor, filterLdx++, dim, dlt);
196  }
197 
198  if (auto binOp = a.dyn_cast<AffineBinaryOpExpr>()) {
199  // We do not set dim level format for affine expresssion like d0 + d1 on
200  // either loop index at d0 or d1.
201  // We continue the recursion merely to check whether current affine is
202  // admissible or not.
203  return findAffine(merger, tensor, dim, binOp.getLHS(), dlt, filterLdx,
204  false) &&
205  findAffine(merger, tensor, dim, binOp.getRHS(), dlt, filterLdx,
206  false);
207  }
208  // Falls through when it is a constant Affine
209  return true;
210  }
211  default:
212  return false;
213  }
214 }

216 /// Get the total number of compound affine expressions in affineMap that are
217 /// attached to the given tensor. For the following inputs:
218 ///
219 /// affineMap = (d0, d1, d2) => (d0 + d1, d2)
220 /// tensor = ["compressed", "compressed"]
221 ///
222 /// Returns 1 (because the first level is compressed and its corresponding
223 /// affineMap is d0 + d1)
224 static unsigned getNumCompoundAffineOnSparseDims(AffineMap affineMap,
225  Value tensor) {
226  unsigned num = 0;
227  auto enc = getSparseTensorEncoding(tensor.getType());
228  if (enc) {
229  ArrayRef<AffineExpr> exps = affineMap.getResults();
230  for (unsigned rank = 0; rank < exps.size(); rank++) {
231  auto aidx = toOrigDim(enc, rank);
232  auto affine = exps[aidx];
233  if (!affine.isa<AffineDimExpr>())
234  if (!isDenseDLT(getDimLevelType(enc, rank)))
235  num++;
236  }
237  }
238 
239  return num;
240 }

252 /// Helper method to inspect sparse encodings in the tensor types.
253 /// Fills the per-dimension sparsity information for all tensors.
254 /// Returns true if the sparse annotations and affine subscript
255 /// expressions of all tensors are admissible. Returns false if
256 /// no annotations are found or inadmissible constructs occur.
257 static bool findSparseAnnotations(CodegenEnv &env) {
258  bool annotated = false;
259  unsigned filterLdx = env.merger().getFilterLoopStartingIdx();
260  for (OpOperand &t : env.op()->getOpOperands()) {
261  auto map = env.op().getMatchingIndexingMap(&t);
262  auto enc = getSparseTensorEncoding(t.get().getType());
263  if (enc)
264  annotated = true;
265  assert(map.getNumResults() == env.op().getRank(&t));
266  for (unsigned d = 0, rank = map.getNumResults(); d < rank; d++) {
267  unsigned tensor = t.getOperandNumber();
268  AffineExpr a = map.getResult(toOrigDim(enc, d));
269  if (!findAffine(env.merger(), tensor, d, a, getDimLevelType(enc, d),
270  filterLdx))
271  return false; // inadmissible affine expression
272  }
273  }
274  assert(filterLdx == env.merger().getNumLoops());
275  return annotated;
276 }

278 /// A helper to compute a topological sort. O(n^2) time complexity
279 /// as we use adj matrix for the graph.
280 /// The sorted result will put the first Reduction iterator to the
281 /// latest possible index.
282 static bool topSortOptimal(CodegenEnv &env, unsigned n,
283  ArrayRef<utils::IteratorType> iteratorTypes,
284  std::vector<unsigned> &inDegree,
285  std::vector<std::vector<bool>> &adjM) {
286  std::vector<unsigned> redIt; // reduce iterator with 0 degree
287  std::vector<unsigned> parIt; // parallel iterator with 0 degree
288  std::vector<unsigned> filterIt; // filter loop with 0 degree
289  for (unsigned i = 0; i < n; i++) {
290  if (inDegree[i] == 0) {
291  if (env.merger().isFilterLoop(i))
292  filterIt.push_back(i);
293  else if (linalg::isReductionIterator(iteratorTypes[i]))
294  redIt.push_back(i);
295  else
296  parIt.push_back(i);
297  }
298  }
299 
300  while (!redIt.empty() || !parIt.empty() || !filterIt.empty()) {
301  // We always choose in order of filter loop -> parallel loop -> reduction
302  // loop because
303  // 1. Putting reduction loop early might make the loop sequence
304  // inadmissible.
305  // 2. Filter loops should be put as early as possible for better
306  // performance, since only one (if any) iteration will carry the
307  // computation. E.g., for (1 to N)
308  // for (1 to M)
309  // for (1 to K)
310  // if (xxx)
311  // O(X) computation => O(NMK+NMX) time complexity
312  //
313  // By putting the filter loop one level up, we got
314  //
315  // for (1 to N)
316  // for (1 to K)
317  // if (xxx)
318  // for (1 to M)
319  // O(X) computation => O(NK+NMX) time complexity
320  auto &it = !filterIt.empty() ? filterIt : (!parIt.empty() ? parIt : redIt);
321  auto src = it.back();
322  env.topSortPushBack(src);
323  it.pop_back();
324  // Update in-degree, and push 0-degree node into worklist.
325  for (unsigned dst = 0; dst < n; dst++) {
326  if (adjM[src][dst] && --inDegree[dst] == 0) {
327  if (env.merger().isFilterLoop(dst))
328  filterIt.push_back(dst);
329  else if (linalg::isReductionIterator(iteratorTypes[dst]))
330  redIt.push_back(dst);
331  else
332  parIt.push_back(dst);
333  }
334  }
335  }
336  return env.topSortSize() == n;
337 }

339 /// Helper method to add all constraints from the indices in one affine
340 /// expression before all indices in the other affine expression. For
341 /// example i0+i1 < i2+i3+1 yields i0<i2, i0<i3, i1<i2, and i1<i3.
342 /// The affine expression `a` is empty iff `fidx` have a value, leading to
343 /// b = (i0 + i1) < fidx => i0 < fidx, i1 < fidx.
344 /// The affine expression `b` is empty iff `tidx` have a value, leading to
345 /// tidx < a = (i0 + i1) => tidx < i0, tidx < i1.
346 static void addAffineOrderings(std::vector<std::vector<bool>> &adjM,
347  std::vector<unsigned> &inDegree, AffineExpr a,
348  AffineExpr b, Optional<unsigned> fidx,
349  Optional<unsigned> tidx) {
350  if (!a && !b) {
351  // Recursion leaf.
352  assert(fidx && tidx);
353  unsigned f = *fidx, t = *tidx;
354  if (!adjM[f][t]) {
355  adjM[f][t] = true;
356  inDegree[t]++;
357  }
358  return;
359  }
360  // Picks an affine expression and expand (recurse into) it.
361  auto toExpand = a ? a : b;
362  switch (toExpand.getKind()) {
363  case AffineExprKind::DimId: {
364  auto idx = toExpand.cast<AffineDimExpr>().getPosition();
365  if (toExpand == a)
366  addAffineOrderings(adjM, inDegree, AffineExpr(), b, idx, tidx);
367  else // toExpand == b
368  addAffineOrderings(adjM, inDegree, a, AffineExpr(), fidx, idx);
369  break;
370  }
371  case AffineExprKind::Add:
372  case AffineExprKind::Mul: {
373  auto binOp = toExpand.cast<AffineBinaryOpExpr>();
374  if (toExpand == a) {
375  addAffineOrderings(adjM, inDegree, binOp.getLHS(), b, fidx, tidx);
376  addAffineOrderings(adjM, inDegree, binOp.getRHS(), b, fidx, tidx);
377  } else {
378  addAffineOrderings(adjM, inDegree, a, binOp.getLHS(), fidx, tidx);
379  addAffineOrderings(adjM, inDegree, a, binOp.getRHS(), fidx, tidx);
380  }
381  break;
382  }
383  default:
384  break;
385  }
386 }

388 static void tryLoosenAffineDenseConstraints(linalg::GenericOp op,
389  Optional<unsigned> &fldx,
390  AffineExpr &fa,
391  Optional<unsigned> &tldx,
392  AffineExpr &ta) {
393  // We use a heuristic here to only pick one dim expression from each
394  // compound affine expression to establish the order between two dense
395  // dimensions.
396  if (!tldx) {
397  AffineDimFinder finder(op);
398  // NOTE: The ordering can only be loosen when the destination level is
399  // dense (when !tldx), for [dense, sparse] -> (d0 + d1, d2), we still
400  // require both d0 < d2 and d1 < d2 to ensure correct ordering (i.e.,
401  // no ordering like d0->d2->d1).
402  // TODO: this is obviously a sub optimal solution.
403  if (!fldx && !fa.isa<AffineConstantExpr>()) {
404  // Heuristic: we prefer parallel loop for lhs to reduce the chance
405  // we add reduce < parallel ordering.
406  finder.setPickedIterType(utils::IteratorType::parallel);
407  finder.walkPostOrder(fa);
408  fa = finder.getDimExpr();
409  fldx = finder.getDimExpr().getPosition();
410  }
411  if (!ta.isa<AffineConstantExpr>()) {
412  // Heuristic: we prefer reduction loop for rhs to reduce the chance
413  // addint reduce < parallel ordering.
414  finder.setPickedIterType(utils::IteratorType::reduction);
415  finder.walkPostOrder(ta);
416  ta = finder.getDimExpr();
417  tldx = finder.getDimExpr().getPosition();
418  }
419  }
420 }

422 /// Computes a topologically sorted iteration graph for the linalg
423 /// operation. Ensures all tensors are visited in natural index order. This
424 /// is essential for sparse storage formats since these only support access
425 /// along fixed dimensions. Even for dense storage formats, however, the
426 /// natural index order yields innermost unit-stride access with better
427 /// spatial locality.
428 static bool computeIterationGraph(CodegenEnv &env, unsigned mask,
429  OpOperand *skip = nullptr) {
430  // Set up an n x n from/to adjacency matrix of the iteration graph
431  // for the implicit loop indices i_0 .. i_n-1.
432  unsigned n = env.merger().getNumLoops();
433  std::vector<std::vector<bool>> adjM(n, std::vector<bool>(n, false));
434  std::vector<unsigned> inDegree(n, 0); // in-degree of each node.
435  auto iteratorTypes = env.op().getIteratorTypesArray();
436  // Iterate over the indexing maps of every tensor in the tensor expression.
437  for (OpOperand &t : env.op()->getOpOperands()) {
438  // Get map and encoding.
439  auto map = env.op().getMatchingIndexingMap(&t);
440  auto enc = getSparseTensorEncoding(t.get().getType());
441  assert(map.getNumDims() + getNumCompoundAffineOnSparseDims(env.op()) == n);
442  // Skip dense tensor constraints when not requested.
443  if (!(mask & SortMask::kIncludeDense) && !enc)
444  continue;
445  // Each tensor expression and optional dimension ordering (row-major
446  // by default) puts an ordering constraint on the loop indices. For
447  // example, the tensor expresion A_ijk forces the ordering i < j < k
448  // on the loop indices if no explicit dimension ordering is given.
449  for (unsigned d = 0, rank = map.getNumResults(); d < rank; d++) {
450  AffineExpr ta = map.getResult(toOrigDim(enc, d));
451  Optional<unsigned> tldx =
452  env.merger().getLoopIdx(t.getOperandNumber(), d);
453 
454  // Filter loops should be constructed after all the dependent loops,
455  // i.e., d0 + d1 < filter_loop(d0 + d1)
456  if (tldx && env.merger().isFilterLoop(*tldx)) {
457  assert(!ta.isa<AffineDimExpr>() &&
458  !isDenseDLT(getDimLevelType(enc, d)));
459  addAffineOrderings(adjM, inDegree, ta, AffineExpr(), std::nullopt,
460  tldx);
461  // Now that the ordering of affine expression is captured by filter
462  // loop idx, we only need to ensure the affine ordering against filter
463  // loop. Thus, we reset the affine express to nil here to mark it as
464  // resolved.
465  ta = AffineExpr();
466  }
467 
468  // Skip tensor during cycle resolution, though order between filter loop
469  // and dependent loops need to be guaranteed unconditionally.
470  if (&t == skip)
471  continue;
472 
473  if (d > 0) {
474  AffineExpr fa = map.getResult(toOrigDim(enc, d - 1));
475  Optional<unsigned> fldx =
476  env.merger().getLoopIdx(t.getOperandNumber(), d - 1);
477 
478  // Applying order constraints on every pair of dimExpr between two
479  // compound affine expressions can sometime too strict:
480  // E.g, for [dense, dense] -> (d0 + d1, d2 + d3).
481  // It is totally fine to have loop sequence d0->d2->d1->d3 instead of
482  // requiring d0 < d2, d1 < d2, d0 < d3, d1 < d3.
483  if (!(mask & SortMask::kIncludeDense))
484  tryLoosenAffineDenseConstraints(env.op(), fldx, fa, tldx, ta);
485 
486  // (d0 + d1) < (d2 + d3), or
487  // filter_loop_d-1 < (d2 + d3), or
488  // (d0 + d1) < filter_loop_d, or
489  // filter_loop_d-1 < filter_loop_d depending on whether fa/ta is reset
490  // above.
491  addAffineOrderings(adjM, inDegree, fa, ta, fldx, tldx);
492  }
493  }
494  // Push unrelated loops into sparse iteration space, so these
495  // will be skipped more often.
496  if (mask & SortMask::kIncludeUndef) {
497  unsigned tensor = t.getOperandNumber();
498  for (unsigned i = 0; i < n; i++)
499  if (isCompressedDLT(env.dlt(tensor, i)) ||
500  isSingletonDLT(env.dlt(tensor, i))) {
501  for (unsigned j = 0; j < n; j++)
502  if (isUndefDLT(env.dlt(tensor, j))) {
503  adjM[i][j] = true;
504  inDegree[j]++;
505  }
506  } else {
507  assert(isDenseDLT(env.dlt(tensor, i)) ||
508  isUndefDLT(env.dlt(tensor, i)));
509  }
510  }
511  }
512  // Topologically sort the iteration graph to determine loop order.
513  // Report failure for a cyclic iteration graph.
514  env.topSortClear(n);
515  return topSortOptimal(env, n, iteratorTypes, inDegree, adjM);
516 }

524 /// Returns true when the tensor expression is admissible for codegen.
525 /// Since all sparse input tensors are admissible, we just need to check
526 /// whether the out tensor in the tensor expression codegen is admissible.
527 /// Sets `sparseOut` to the tensor and `outerParNest` to the outer injective
528 /// nesting depth when a "truly dynamic" sparse tensor output occurs.
529 static bool isAdmissibleTensorExp(CodegenEnv &env, unsigned exp,
530  OpOperand **sparseOut,
531  unsigned *outerParNest) {
532  // We reject any expression that makes a reduction from `-outTensor`, as those
533  // expressions create a dependency between the current iteration (i) and the
534  // previous iteration (i-1). It would require iterating over the whole
535  // coordinate space, which prevent exploiting sparsity for faster code.
536  for (utils::IteratorType it : env.op().getIteratorTypesArray()) {
537  if (it == utils::IteratorType::reduction) {
538  if (env.merger().hasNegateOnOut(exp))
539  return false;
540  break;
541  }
542  }
543 
544  OpOperand *lhs = env.op().getDpsInitOperand(0);
545  unsigned tensor = lhs->getOperandNumber();
546  auto enc = getSparseTensorEncoding(lhs->get().getType());
547  // An non-annotated output tensor is assumed dense, and becomes a random
548  // access n-dim memref. Admissible since insertions cannot occur.
549  if (!enc)
550  return true;
551  // An all-dense annotated "sparse" output tensor becomes a linearized random
552  // access 1-dim memref. Also admissible since insertions cannot occur.
553  bool allDense = true;
554  unsigned numLoops =
555  env.merger().getNumLoops(); // numNativeLoops + numFilterLoops
556  for (unsigned i = 0; i < env.merger().getNumLoops(); i++)
557  if (isCompressedDLT(env.dlt(tensor, i)) ||
558  isSingletonDLT(env.dlt(tensor, i))) {
559  allDense = false;
560  break;
561  } else {
562  assert(isDenseDLT(env.dlt(tensor, i)) || isUndefDLT(env.dlt(tensor, i)));
563  }
564  if (allDense)
565  return true;
566 
567  // TODO: support compound affine expression on sparse output.
568  if (getNumCompoundAffineOnSparseDims(env.op().getMatchingIndexingMap(lhs),
569  lhs->get()) != 0)
570  return false;
571 
572  // A tensor expression with a sparse output tensor that changes its values
573  // but not its nonzero structure, an operation called "simply dynamic" in
574  // [Bik96,Ch9], is also admissible without special env.
575  if (env.merger().isSingleCondition(tensor, exp))
576  return true;
577 
578  // Accept "truly dynamic" if the output tensor materializes uninitialized
579  // into the computation and insertions occur in lexicographic index order.
580  if (isMaterializing(lhs->get())) {
581  auto iteratorTypes = env.op().getIteratorTypesArray();
582  unsigned nest = 0;
583  for (unsigned i = 0; i < numLoops; i++) {
584  if (!env.merger().isFilterLoop(env.topSortAt(i))) {
585  // We only count non-filter loops as filter loops should be considered
586  // as a special type of parallel loops.
587  if (linalg::isReductionIterator(iteratorTypes[env.topSortAt(i)]))
588  break; // terminate at first reduction
589  nest++;
590  }
591  }
592  // Determine admissible dynamic insertion situations:
593  // (1) fully injective, since there are no reductions,
594  // (2) admissible 1-d expansion in innermost dimension.
595  if (nest >= env.op().getRank(lhs) - 1) {
596  *sparseOut = lhs;
597  *outerParNest = nest;
598  return true;
599  }
600  }
601  return false;
602 }

608 /// Local bufferization of all dense and sparse data structures.
609 static void genBuffers(CodegenEnv &env, OpBuilder &builder) {
610  linalg::GenericOp op = env.op();
611  Location loc = op.getLoc();
612  assert(op.getNumOperands() == op.getNumDpsInputs() + 1);
613 
614  env.emitter().initializeLoopEmit(
615  builder, loc,
616  /// Generates buffer for the output tensor.
617  /// Note that all sparse kernels assume that when all elements are written
618  /// to (viz. x(i) = y(i) * z(i)), the output buffer is already initialized
619  /// to all zeroes and only nonzeroes values are computed and written out.
620  /// For updates (viz. x(i) += y(i) * z(i)), only nonzeroes values are used
621  /// for the updates and no assumption on the original contents of the
622  /// output buffer is necessary.
623  [&op](OpBuilder &builder, Location loc, Value memref,
624  Value tensor) -> Value {
625  // Must not be a sparse tensor.
626  assert(!getSparseTensorEncoding(tensor.getType()));
627  // Two output tensor references should point to the same object.
628  OpOperand *lhs = op.getDpsInitOperand(0);
629  assert(lhs->get() == tensor);
630  // An output tensor can simply materialize from the buffer of the tensor
631  // that appears in the outs() clause. For updates, this has the
632  // advantage that only the nonzero value are involved in the
633  // computation, keeping the operation O(nnz). In all other cases, we are
634  // forced to zero out the buffer to enforce the assumption above, which
635  // may negatively impact running complexity (viz. O(n^2 + nnz) vs.
636  // O(nnz) for matrices).
637  // TODO: use better analysis to avoid zeroing out the buffer?
638  bool isInit = op.isInitTensor(lhs);
639  Value init = memref;
640  if (!isInit) {
641  Value zero = constantZero(builder, loc,
642  getElementTypeOrSelf(tensor.getType()));
643  builder.create<linalg::FillOp>(loc, ValueRange{zero},
644  ValueRange{init});
645  }
646  return init;
647  });
648 }

714 /// Generates insertion code to implement dynamic tensor store.
715 static void genInsertionStore(CodegenEnv &env, OpBuilder &builder, OpOperand *t,
716  Value rhs) {
717  linalg::GenericOp op = env.op();
718  Location loc = op.getLoc();
719  // Direct insertion in lexicographic index order.
720  if (!env.isExpand()) {
721  unsigned rank = op.getRank(t);
722  SmallVector<Value> indices;
723  for (unsigned i = 0; i < rank; i++) {
724  assert(env.emitter().getLoopIV(i));
725  indices.push_back(env.emitter().getLoopIV(i));
726  }
727  Value chain = env.getInsertionChain();
728  env.updateInsertionChain(
729  builder.create<InsertOp>(loc, rhs, chain, indices));
730  return;
731  }
732  // Generates insertion code along expanded access pattern.
733  // if (!expFilled[i]) then
734  // expFilled[i] = true
735  // expAdded[inserts++] = i
736  // endif
737  // values[i] = rhs
738  Value values = env.getExpandValues();
739  Value filled = env.getExpandFilled();
740  Value added = env.getExpandAdded();
741  Value count = env.getExpandCount();
742  Value index = genIndex(env, t);
743  Value fval = constantI1(builder, loc, false);
744  Value tval = constantI1(builder, loc, true);
745  // If statement.
746  Value isFilled = builder.create<memref::LoadOp>(loc, filled, index);
747  Value cond = builder.create<arith::CmpIOp>(loc, arith::CmpIPredicate::eq,
748  isFilled, fval);
749  scf::IfOp ifOp = builder.create<scf::IfOp>(loc, builder.getIndexType(), cond,
750  /*else=*/true);
751  // True branch.
752  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
753  builder.create<memref::StoreOp>(loc, tval, filled, index);
754  builder.create<memref::StoreOp>(loc, index, added, count);
755  Value one = constantIndex(builder, loc, 1);
756  Value add = builder.create<arith::AddIOp>(loc, count, one);
757  builder.create<scf::YieldOp>(loc, add);
758  // False branch.
759  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
760  builder.create<scf::YieldOp>(loc, count);
761  builder.setInsertionPointAfter(ifOp);
762  // Value assignment.
763  env.updateExpandCount(ifOp.getResult(0));
764  builder.create<memref::StoreOp>(loc, rhs, values, index);
765 }

767 /// Generates a load on a dense or sparse tensor.
768 static Value genTensorLoad(CodegenEnv &env, OpBuilder &builder, unsigned exp) {
769  // Test if the load was hoisted to a higher loop nest.
770  Value val = env.exp(exp).val;
771  if (val)
772  return val;
773 
774  // Load during insertion.
775  linalg::GenericOp op = env.op();
776  OpOperand *t = &op->getOpOperand(env.exp(exp).tensor);
777  if (env.isSparseOutput(t)) {
778  if (env.isCustomReduc())
779  return genInsertionLoadReduce(env, builder, t);
780  return genInsertionLoad(env, builder, t);
781  }
782  // Actual load.
783  SmallVector<Value> args;
784  Value ptr = genSubscript(env, builder, t, args);
785  return builder.create<memref::LoadOp>(op.getLoc(), ptr, args);
786 }

788 /// Generates a store on a dense or sparse tensor.
789 static void genTensorStore(CodegenEnv &env, OpBuilder &builder, unsigned exp,
790  Value rhs) {
791  linalg::GenericOp op = env.op();
792  Location loc = op.getLoc();
793  // Test if this is a scalarized reduction.
794  if (env.isReduc()) {
795  env.updateReduc(rhs);
796  return;
797  }
798  // Store during insertion.
799  OpOperand *t = op.getDpsInitOperand(0);
800  if (env.isSparseOutput(t)) {
801  if (!rhs) {
802  // Only unary and binary are allowed to return uninitialized rhs
803  // to indicate missing output.
804  assert(env.exp(exp).kind == kUnary || env.exp(exp).kind == kBinary);
805  } else if (env.exp(exp).kind == kSelect) {
806  // Select operation insertion.
807  Value chain = env.getInsertionChain();
808  scf::IfOp ifOp =
809  builder.create<scf::IfOp>(loc, chain.getType(), rhs, /*else=*/true);
810  builder.setInsertionPointToStart(&ifOp.getThenRegion().front());
811  // Existing value was preserved to be used here.
812  assert(env.exp(exp).val);
813  Value v0 = env.exp(exp).val;
814  genInsertionStore(env, builder, t, v0);
815  env.exp(exp).val = Value();
816  // Yield modified insertion chain along true branch.
817  Value mchain = env.getInsertionChain();
818  builder.create<scf::YieldOp>(op.getLoc(), mchain);
819  // Yield original insertion chain along false branch.
820  builder.setInsertionPointToStart(&ifOp.getElseRegion().front());
821  builder.create<scf::YieldOp>(loc, chain);
822  // Done with if statement.
823  env.updateInsertionChain(ifOp->getResult(0));
824  builder.setInsertionPointAfter(ifOp);
825  } else {
826  genInsertionStore(env, builder, t, rhs);
827  }
828  return;
829  }
830  // Actual store.
831  SmallVector<Value> args;
832  Value ptr = genSubscript(env, builder, t, args);
833  builder.create<memref::StoreOp>(loc, rhs, ptr, args);
834 }

846 /// Semi-ring branches are simply inlined by the sparse compiler. Prior
847 /// analysis has verified that all computations are "local" to the inlined
848 /// branch or otherwise invariantly defined outside the loop nest, with the
849 /// exception of index computations, which need to be relinked to actual
850 /// inlined cloned code.
851 static Value relinkBranch(CodegenEnv &env, RewriterBase &rewriter, Block *block,
852  Value e, unsigned ldx) {
853  if (Operation *def = e.getDefiningOp()) {
854  if (auto indexOp = dyn_cast<linalg::IndexOp>(def))
855  return genIndexValue(env, indexOp.getDim());
856  if (def->getBlock() == block) {
857  for (unsigned i = 0, n = def->getNumOperands(); i < n; i++)
858  def->setOperand(
859  i, relinkBranch(env, rewriter, block, def->getOperand(i), ldx));
860  }
861  }
862  return e;
863 }

865 /// Recursively generates tensor expression.
866 static Value genExp(CodegenEnv &env, RewriterBase &rewriter, unsigned exp,
867  unsigned ldx) {
868  linalg::GenericOp op = env.op();
869  Location loc = op.getLoc();
870 
871  if (exp == -1u)
872  return Value();
873  if (env.exp(exp).kind == Kind::kTensor)
874  return genTensorLoad(env, rewriter, exp);
875  if (env.exp(exp).kind == Kind::kInvariant)
876  return genInvariantValue(env, exp);
877  if (env.exp(exp).kind == Kind::kIndex)
878  return genIndexValue(env, env.exp(exp).index);
879 
880  if (env.exp(exp).kind == Kind::kReduce)
881  env.startCustomReduc(exp); // enter custom
882 
883  Value v0 = genExp(env, rewriter, env.exp(exp).children.e0, ldx);
884  Value v1 = genExp(env, rewriter, env.exp(exp).children.e1, ldx);
885  Value ee = env.merger().buildExp(rewriter, loc, exp, v0, v1);
886  if (ee && (env.exp(exp).kind == Kind::kUnary ||
887  env.exp(exp).kind == Kind::kBinary ||
888  env.exp(exp).kind == Kind::kBinaryBranch ||
889  env.exp(exp).kind == Kind::kReduce ||
890  env.exp(exp).kind == Kind::kSelect))
891  ee = relinkBranch(env, rewriter, ee.getParentBlock(), ee, ldx);
892 
893  if (env.exp(exp).kind == Kind::kReduce)
894  env.endCustomReduc(); // exit custom
895 
896  if (env.exp(exp).kind == kSelect) {
897  assert(!env.exp(exp).val);
898  env.exp(exp).val = v0; // Preserve value for later use.
899  }
900 
901  return ee;
902 }

904 /// Hoists loop invariant tensor loads for which indices have been exhausted.
905 static void genInvariants(CodegenEnv &env, OpBuilder &builder, unsigned exp,
906  unsigned ldx, bool atStart) {
907  if (exp == -1u)
908  return;
909  if (env.exp(exp).kind == Kind::kTensor) {
910  // Inspect tensor indices.
911  bool atLevel = ldx == -1u;
912  linalg::GenericOp op = env.op();
913  OpOperand &t = op->getOpOperand(env.exp(exp).tensor);
914  auto map = op.getMatchingIndexingMap(&t);
915  auto enc = getSparseTensorEncoding(t.get().getType());
916  for (unsigned d = 0, rank = map.getNumResults(); d < rank; d++) {
917  AffineExpr a = map.getResult(toOrigDim(enc, d));
918  Optional<unsigned> sldx =
919  env.merger().getLoopIdx(t.getOperandNumber(), d);
920  if (sldx && env.merger().isFilterLoop(*sldx)) {
921  if (!env.getLoopIdxValue(*sldx))
922  // The filter loops has not been constructed.
923  return;
924  if (*sldx == ldx)
925  atLevel = true;
926  } else if (!isInvariantAffine(env, a, ldx, atLevel))
927  return; // still in play
928  }
929  // All exhausted at this level (atLevel denotes exactly at this level).
930  if (!atLevel)
931  return;
932  OpOperand *lhs = op.getDpsInitOperand(0);
933  if (lhs == &t) {
934  // Start or end a scalarized reduction
935  if (atStart) {
936  Value load = env.isCustomReduc() ? env.getCustomRedId()
937  : genTensorLoad(env, builder, exp);
938  env.startReduc(exp, load);
939  } else {
940  genTensorStore(env, builder, exp, env.endReduc());
941  }
942  } else {
943  // Start or end loop invariant hoisting of a tensor load.
944  env.exp(exp).val = atStart ? genTensorLoad(env, builder, exp) : Value();
945  }
946  } else if (env.exp(exp).kind != Kind::kInvariant &&
947  env.exp(exp).kind != Kind::kIndex) {
948  // Traverse into the binary operations. Note that we only hoist
949  // tensor loads, since subsequent MLIR/LLVM passes know how to
950  // deal with all other kinds of derived loop invariants.
951  if (env.exp(exp).kind == Kind::kReduce)
952  env.startCustomReduc(exp); // enter custom
953  unsigned e0 = env.exp(exp).children.e0;
954  unsigned e1 = env.exp(exp).children.e1;
955  genInvariants(env, builder, e0, ldx, atStart);
956  genInvariants(env, builder, e1, ldx, atStart);
957  if (env.exp(exp).kind == Kind::kReduce)
958  env.endCustomReduc(); // exit custom
959  }
960 }

962 /// Generates an expanded access pattern in innermost dimension.
963 static void genExpand(CodegenEnv &env, OpBuilder &builder, unsigned at,
964  bool atStart) {
965  linalg::GenericOp op = env.op();
966  OpOperand *lhs = op.getDpsInitOperand(0);
967  if (!env.atExpandLevel(lhs, op.getRank(lhs), at))
968  return; // not needed at this level
969  assert(!env.isReduc());
970  // Generate start or end of an expanded access pattern. Note that because
971  // an expension does not rely on the ongoing contents of the sparse storage
972  // scheme, we can use the original tensor as incoming SSA value (which
973  // simplifies codegen a bit). If expansion on the actual contents is ever
974  // needed, we will need to use the SSA value in the insertion chain instead.
975  Value tensor = lhs->get();
976  Location loc = op.getLoc();
977  if (atStart) {
978  auto dynShape = {ShapedType::kDynamic};
979  Type etp = tensor.getType().cast<ShapedType>().getElementType();
980  Type t1 = MemRefType::get(dynShape, etp);
981  Type t2 = MemRefType::get(dynShape, builder.getI1Type());
982  Type t3 = MemRefType::get(dynShape, builder.getIndexType());
983  Type t4 = builder.getIndexType();
984  auto r = builder.create<ExpandOp>(loc, TypeRange({t1, t2, t3, t4}), tensor);
985  assert(r.getNumResults() == 4);
986  env.startExpand(r.getResult(0), r.getResult(1), r.getResult(2),
987  r.getResult(3));
988  } else {
989  SmallVector<Value> indices;
990  for (unsigned i = 0; i < at; i++)
991  indices.push_back(env.emitter().getLoopIV(i));
992  Value values = env.getExpandValues();
993  Value filled = env.getExpandFilled();
994  Value added = env.getExpandAdded();
995  Value count = env.getExpandCount();
996  Value chain = env.getInsertionChain();
997  Value compress = builder.create<CompressOp>(loc, values, filled, added,
998  count, chain, indices);
999  env.updateInsertionChain(compress);
1000  env.endExpand();
1001  }

1004 /// Returns parallelization strategy. Any implicit loop in the Linalg
1005 /// operation that is marked "parallel" is a candidate. Whether it is actually
1006 /// converted to a parallel operation depends on the requested strategy.
1007 static bool isParallelFor(CodegenEnv &env, bool isOuter, bool isSparse) {
1008  // Reject parallelization of sparse output.
1009  if (env.hasSparseOutput())
1010  return false;
1011  // Parallel loops on tensor expansion can cause data races.
1012  if (env.isExpand())
1013  return false;
1014  // Inspect strategy.
1015  switch (env.options().parallelizationStrategy) {
1016  case SparseParallelizationStrategy::kNone:
1017  return false;
1018  case SparseParallelizationStrategy::kDenseOuterLoop:
1019  return isOuter && !isSparse;
1020  case SparseParallelizationStrategy::kAnyStorageOuterLoop:
1021  return isOuter;
1022  case SparseParallelizationStrategy::kDenseAnyLoop:
1023  return !isSparse;
1024  case SparseParallelizationStrategy::kAnyStorageAnyLoop:
1025  return true;
1026  }
1027  llvm_unreachable("unexpected parallelization strategy");
1028 }

1030 /// Generates a for-loop on a single index.
1031 static Operation *genFor(CodegenEnv &env, OpBuilder &builder, bool isOuter,
1032  bool isInner, unsigned idx, ArrayRef<size_t> tids,
1033  ArrayRef<size_t> dims) {
1034  linalg::GenericOp op = env.op();
1035  Location loc = op.getLoc();
1036  auto iteratorTypes = op.getIteratorTypesArray();
1037  bool isSparse = llvm::any_of(tids, [idx, &env](size_t tid) {
1038  return isCompressedDLT(env.dlt(tid, idx)) ||
1039  isSingletonDLT(env.dlt(tid, idx));
1040  });
1041 
1042  bool isParallel = isParallelFor(env, isOuter, isSparse);
1043 
1044  Operation *loop = *env.genLoopBoundary([&](MutableArrayRef<Value> reduc) {
1045  if (env.merger().isFilterLoop(idx)) {
1046  size_t tid = tids.front(), dim = dims.front();
1047  // tids/dims must only have one value because filter loops only
1048  // corresponding to the one and only sparse tensor level.
1049  assert(isSparse && tids.size() == 1 && dims.size() == 1);
1050  OpOperand *t = &op->getOpOperand(tid);
1051  auto enc = getSparseTensorEncoding(t->get().getType());
1052  // Retrieves the affine expression for the filter loop.
1053  AffineExpr a =
1054  op.getMatchingIndexingMap(t).getResult(toOrigDim(enc, dim));
1055  return env.emitter().enterFilterLoopOverTensorAtDim(builder, loc, tid,
1056  dim, a, reduc);
1057  }
1058  return env.emitter().enterLoopOverTensorAtDim(builder, loc, tids, dims,
1059  reduc, isParallel);
1060  });
1061  assert(loop);
1062  return loop;
1063 }

1094 /// Generates the induction structure for a while-loop.
1095 static void finalizeWhileOp(CodegenEnv &env, OpBuilder &builder, unsigned idx,
1096  bool needsUniv, BitVector &induction,
1097  scf::WhileOp whileOp) {
1098  Location loc = env.op().getLoc();
1099  // Finalize each else branch of all if statements.
1100  if (env.isReduc() || env.isExpand() || env.getInsertionChain()) {
1101  while (auto ifOp = dyn_cast_or_null<scf::IfOp>(
1102  builder.getInsertionBlock()->getParentOp())) {
1103  unsigned y = 0;
1104  SmallVector<Value> yields;
1105  if (env.isReduc()) {
1106  yields.push_back(env.getReduc());
1107  env.updateReduc(ifOp.getResult(y++));
1108  }
1109  if (env.isExpand()) {
1110  yields.push_back(env.getExpandCount());
1111  env.updateExpandCount(ifOp->getResult(y++));
1112  }
1113  if (env.getInsertionChain()) {
1114  yields.push_back(env.getInsertionChain());
1115  env.updateInsertionChain(ifOp->getResult(y++));
1116  }
1117  assert(y == yields.size());
1118  builder.create<scf::YieldOp>(loc, yields);

1513  PatternRewriter &rewriter) const override {

1515  if (op.getNumDpsInits() != 1)
1516  return failure();

1519  unsigned numTensors = op->getNumOperands();
1520  unsigned numLoops = op.getNumLoops();

1522  CodegenEnv env(op, options, numTensors, numLoops, numFilterLoops);
1523

1525  // information for all tensors to loop indices in the kernel.

1530  Optional<unsigned> optExp = env.merger().buildTensorExpFromLinalg(op);
1531  if (!optExp)

1536  // are visited in natural index order. Gradually relaxes the considered
1537  // constraints until an acyclic iteration graph results, such that sparse
1538  // code generation can proceed. As a last resort, an attempt is made
1539  // to resolve cycles by inserting a conversion.
1540  bool isAdmissible = false;
1541  bool hasCycle = true;

1552  isAdmissible = true;

1556  }

1576  // iteration graph results.
