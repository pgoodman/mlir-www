25 class SliceConverter : public OpRewritePattern<tosa::SliceOp> {
26 public:
27  using OpRewritePattern<tosa::SliceOp>::OpRewritePattern;
28 
29  LogicalResult matchAndRewrite(tosa::SliceOp sliceOp,
30  PatternRewriter &rewriter) const final {
31  Location loc = sliceOp.getLoc();
32  Value input = sliceOp.getInput();
33  SmallVector<int64_t> strides, sizes, starts;
34  starts = extractFromI64ArrayAttr(sliceOp.getStart());
35  strides.resize(sliceOp.getType().template cast<ShapedType>().getRank(), 1);
36 
37  SmallVector<Value> dynSizes;
38  for (const auto &i : llvm::enumerate(sliceOp.getSize())) {
39  int64_t size = i.value().cast<IntegerAttr>().getInt();
40  size_t index = i.index();
41  sizes.push_back(size == -1 ? ShapedType::kDynamic : size);
42  if (!ShapedType::isDynamic(sizes.back()))
43  continue;
44 
45  auto dim = rewriter.create<tensor::DimOp>(loc, input, index);
46  auto offset = rewriter.create<arith::ConstantOp>(
47  loc, rewriter.getIndexAttr(starts[index]));
48  dynSizes.push_back(rewriter.create<arith::SubIOp>(loc, dim, offset));
49  }
50 
51  auto newSliceOp = rewriter.create<tensor::ExtractSliceOp>(
52  sliceOp.getLoc(), sliceOp.getType(), input, ValueRange({}), dynSizes,
53  ValueRange({}), rewriter.getDenseI64ArrayAttr(starts),
54  rewriter.getDenseI64ArrayAttr(sizes),
55  rewriter.getDenseI64ArrayAttr(strides));
56 
57  rewriter.replaceOp(sliceOp, newSliceOp.getResult());
58  return success();
59  }
60 };

62 class PadConverter : public OpRewritePattern<tosa::PadOp> {
63 public:
64  using OpRewritePattern<tosa::PadOp>::OpRewritePattern;
65 
66  LogicalResult matchAndRewrite(tosa::PadOp padOp,
67  PatternRewriter &rewriter) const final {
68  auto loc = padOp.getLoc();
69  auto input = padOp.getInput1();
70  auto padding = padOp.getPadding();
71 
72  ShapedType inputTy = input.getType().cast<ShapedType>();
73  Type elementTy = inputTy.getElementType();
74  int64_t rank = inputTy.getRank();
75 
76  // Setup the default constantAttr.
77 
78  Value padConstant;
79 
80  if (padOp.getPadConst()) {
81  padConstant = rewriter.createOrFold<tensor::ExtractOp>(
82  loc, padOp.getPadConst(), ValueRange({}));
83  } else {
84  Attribute constantAttr;
85  if (elementTy.isa<FloatType>()) {
86  constantAttr = rewriter.getFloatAttr(elementTy, 0.0);
87  } else if (elementTy.isa<IntegerType>() && !padOp.getQuantizationInfo()) {
88  constantAttr = rewriter.getIntegerAttr(elementTy, 0);
89  } else if (elementTy.isa<IntegerType>() && padOp.getQuantizationInfo()) {
90  int64_t value = padOp.getQuantizationInfo()->getInputZp();
91  constantAttr = rewriter.getIntegerAttr(elementTy, value);
92  }
93  if (constantAttr)
94  padConstant = rewriter.create<arith::ConstantOp>(loc, constantAttr);
95  }
96 
97  if (!padConstant) {
98  return rewriter.notifyMatchFailure(
99  padOp, "tosa.pad was unable to determine the pad constant value.");
100  }
101 
102  Value lowIndex =
103  rewriter.create<arith::ConstantOp>(loc, rewriter.getIndexAttr(0));
104  Value highIndex =
105  rewriter.create<arith::ConstantOp>(loc, rewriter.getIndexAttr(1));
106 
107  SmallVector<OpFoldResult, 3> lowValues;
108  SmallVector<OpFoldResult, 3> highValues;
109 
110  lowValues.reserve(rank);
111  highValues.reserve(rank);
112 
113  for (int i = 0; i < rank; i++) {
114  Value inputIndex = rewriter.createOrFold<arith::ConstantIndexOp>(loc, i);
115  Value lowVal = rewriter.createOrFold<tensor::ExtractOp>(
116  loc, padding, ValueRange({inputIndex, lowIndex}));
117  Value highVal = rewriter.createOrFold<tensor::ExtractOp>(
118  loc, padding, ValueRange({inputIndex, highIndex}));
119 
120  lowVal = rewriter.createOrFold<arith::IndexCastOp>(
121  loc, rewriter.getIndexType(), lowVal);
122  highVal = rewriter.createOrFold<arith::IndexCastOp>(
123  loc, rewriter.getIndexType(), highVal);
124 
125  lowValues.push_back(lowVal);
126  highValues.push_back(highVal);
127  }
128 
129  auto newPadOp = rewriter.create<tensor::PadOp>(
130  loc, padOp.getType(), input, lowValues, highValues, padConstant);
131 
132  rewriter.replaceOp(padOp, newPadOp.getResult());
133  return success();
134  }
135 };
