37 DiagnosedSilenceableFailure transform::LowerVectorsOp::apply(
38  mlir::transform::TransformResults &transformResults,
39  mlir::transform::TransformState &state) {
40 
41  SmallVector<Operation *> results;
42  ArrayRef<Operation *> payloadOps = state.getPayloadOps(getTarget());
43  for (Operation *target : payloadOps) {
44  // This check can't be part of the verifier because payload IR is
45  // independent from transform IR and may not even exist.
46  if (!target->hasTrait<OpTrait::IsIsolatedFromAbove>()) {
47  return mlir::emitDefiniteFailure(target,
48  "applies only to isolated-from-above "
49  "targets because it needs to apply "
50  "patterns greedily");
51  }
52 
53  MLIRContext *ctx = getContext();
54  RewritePatternSet patterns(ctx);
55  vector::VectorTransposeLowering vectorTransposeLowering =
56  llvm::StringSwitch<vector::VectorTransposeLowering>(
57  getTransposeLowering())
58  .Case("eltwise", vector::VectorTransposeLowering::EltWise)
59  .Case("flat_transpose", vector::VectorTransposeLowering::Flat)
60  .Case("shuffle", vector::VectorTransposeLowering::Shuffle)
61  .Default(vector::VectorTransposeLowering::EltWise);
62  vector::VectorMultiReductionLowering vectorMultiReductionLowering =
63  llvm::StringSwitch<vector::VectorMultiReductionLowering>(
64  getMultireductionLowering())
65  .Case("innerreduction",
66  vector::VectorMultiReductionLowering::InnerReduction)
67  .Default(vector::VectorMultiReductionLowering::InnerParallel);
68  vector::VectorContractLowering vectorContractLowering =
69  llvm::StringSwitch<vector::VectorContractLowering>(
70  getContractionLowering())
71  .Case("matrixintrinsics", vector::VectorContractLowering::Matmul)
72  .Case("dot", vector::VectorContractLowering::Dot)
73  .Case("outerproduct", vector::VectorContractLowering::OuterProduct)
74  .Default(vector::VectorContractLowering::OuterProduct);
75  vector::VectorTransferSplit vectorTransferSplit =
76  llvm::StringSwitch<vector::VectorTransferSplit>(getSplitTransfers())
77  .Case("none", vector::VectorTransferSplit::None)
78  .Case("linalg-copy", vector::VectorTransferSplit::LinalgCopy)
79  .Case("vector-transfers",
80  vector::VectorTransferSplit::VectorTransfer)
81  .Default(vector::VectorTransferSplit::None);
82 
83  vector::VectorTransformsOptions vectorTransformOptions;
84  vectorTransformOptions.setVectorTransformsOptions(vectorContractLowering)
85  .setVectorMultiReductionLowering(vectorMultiReductionLowering)
86  .setVectorTransposeLowering(vectorTransposeLowering)
87  .setVectorTransferSplit(vectorTransferSplit);
88 
89  VectorTransferToSCFOptions vectorTransferToSCFOptions =
90  VectorTransferToSCFOptions().enableFullUnroll(
91  getUnrollVectorTransfers());
92 
93  int maxTransferRank = 1;
94 
95  auto avx2LoweringOptions =
96  x86vector::avx2::LoweringOptions().setTransposeOptions(
97  x86vector::avx2::TransposeLoweringOptions()
98  .lower4x8xf32(getTransposeAvx2Lowering())
99  .lower8x8xf32(getTransposeAvx2Lowering()));
100 
101  vector::populateVectorToVectorCanonicalizationPatterns(patterns);
102 
103  // In the future we may want to more finely select particular stages.
104  // Stage 1: contraction lowerings.
105  patterns.add<mlir::vector::ContractionOpToOuterProductOpLowering,
106  mlir::vector::ContractionOpToMatmulOpLowering,
107  mlir::vector::ContractionOpLowering>(vectorTransformOptions,
108  ctx);
109  vector::populateVectorTransferPermutationMapLoweringPatterns(patterns);
110 
111  // Stage 2: multi-reduction lowerings.
112  vector::populateVectorMultiReductionLoweringPatterns(
113  patterns, vectorTransformOptions.vectorMultiReductionLowering);
114 
115  // Stage 3: Rewrite vector.transfer into full and partial parts.
116  patterns.add<vector::VectorTransferFullPartialRewriter>(
117  ctx, vectorTransformOptions);
118 
119  // Stage 4: Lower vector transfers.
120  vector::populateVectorTransferLoweringPatterns(patterns, maxTransferRank);
121 
122  // Stage 5: Vector to scf patterns.
123  populateVectorToSCFConversionPatterns(
124  patterns, vectorTransferToSCFOptions.setTargetRank(maxTransferRank));
125 
126  // Stage 6: Lower vector.shape_cast.
127  vector::populateVectorShapeCastLoweringPatterns(patterns);
128 
129  // Stage 7: Lower vector.transpose.
130  vector::populateVectorTransposeLoweringPatterns(patterns,
131  vectorTransformOptions);
132  if (getTransposeAvx2Lowering())
133  x86vector::avx2::populateSpecializedTransposeLoweringPatterns(
134  patterns, avx2LoweringOptions, /*benefit=*/10);
135 
136  // Apply everything.
137  if (failed(applyPatternsAndFoldGreedily(target, std::move(patterns))))
138  return DiagnosedSilenceableFailure::definiteFailure();
139 
140  results.push_back(target);
141  }
142 
143  transformResults.set(getResults().cast<OpResult>(), results);
144  return DiagnosedSilenceableFailure::success();
145 }
